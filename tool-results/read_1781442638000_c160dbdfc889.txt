   105→function generate_csrf_token(): string {
   106→    if (empty($_SESSION['csrf_token'])) {
   107→        $_SESSION['csrf_token'] = bin2hex(random_bytes(32));
   108→    }
   109→    return $_SESSION['csrf_token'];
   110→}
   111→
   112→function csrf_field(): string {
   113→    return '<input type="hidden" name="csrf_token" value="' . h(generate_csrf_token()) . '">';
   114→}

   116→function verify_csrf(): bool {
   117→    // Mode test : bypass CSRF
   118→    if (TEST_MODE) return true;
   119→    $token = $_POST['csrf_token'] ?? '';
   120→    return !empty($token) && hash_equals($_SESSION['csrf_token'] ?? '', $token);
   121→}

   708→/**
   709→ * Genere automatiquement un field_name a partir d'un libelle
   710→ * Ex: "Date de prise de poste" → "date_de_prise_de_poste"
   711→ * Ex: "Type d'arrivée" → "type_arrivee"
   712→ */
   713→function generate_field_name(string $label): string {
   714→    // Minuscules
   715→    $name = mb_strtolower($label, 'UTF-8');
   716→    // Supprimer les accents
   717→    $name = transliterator_transliterate('Any-Latin; Latin-ASCII; Lower()', $name);
   718→    if ($name === false) {
   719→        // Fallback manuel si intl pas dispo
   720→        $name = str_replace(
   721→            ['à','â','ä','é','è','ê','ë','ï','î','ô','ö','ù','û','ü','ç','œ','æ','ÿ'],
   722→            ['a','a','a','e','e','e','e','i','i','o','o','u','u','u','c','oe','ae','y'],
   723→            $name
   724→        );
   725→    }
   726→    // Remplacer tout ce qui n'est pas alphanumérique par un underscore
   727→    $name = preg_replace('/[^a-z0-9]+/', '_', $name);
   728→    // Nettoyer les underscores en double et en bordure
   729→    $name = trim($name, '_');
   730→    $name = preg_replace('/_+/', '_', $name);
   731→    return $name ?: 'champ';
   732→}

   734→/**
   735→ * Convertit une liste d'options (une par ligne) en JSON array
   736→ * Ex: "Option A\nOption B" → '["Option A","Option B"]'
   737→ * Si c'est déjà du JSON valide, le retourne tel quel
   738→ */
   739→function parse_options_input(string $input): ?string {
   740→    $input = trim($input);
   741→    if (empty($input)) return null;
   742→    
   743→    // Vérifier si c'est déjà du JSON valide
   744→    $decoded = json_decode($input, true);
   745→    if (json_last_error() === JSON_ERROR_NONE && is_array($decoded)) {
   746→        return $input;
   747→    }
   748→    
   749→    // Traiter comme une liste une option par ligne
   750→    $lines = array_filter(array_map('trim', explode("\n", $input)));
   751→    if (empty($lines)) return null;
   752→    
   753→    return json_encode($lines, JSON_UNESCAPED_UNICODE);
   754→}

   848→/**
   849→ * Déclenche la prochaine étape d'une soumission.
   850→ * Appelé à la création ET après chaque validation.
   851→ *
   852→ * Logique :
   853→ *  - Récupère toutes les étapes du formulaire triées par ordre
   854→ *  - Trouve le plus petit ordre sans tokens générés = prochaine étape
   855→ *  - Si ordre précédent non terminé (séquentiel) → on attend
   856→ *  - Génère les tokens pour tous les destinataires de l'étape courante
   857→ *  - Si plus aucune étape → soumission close
   858→ */
   859→function advance_workflow(int $submission_id): void {
   860→    $pdo = get_pdo();
   861→
   862→    $sub = $pdo->prepare("
   863→        SELECT s.*, f.label as form_label
   864→        FROM submissions s
   865→        JOIN forms f ON f.id = s.form_id
   866→        WHERE s.id = ?
   867→    ");
   868→    $sub->execute([$submission_id]);
   869→    $submission = $sub->fetch(PDO::FETCH_ASSOC);
   870→    if (!$submission || $submission['closed_at']) return;
   871→
   872→    // Toutes les étapes actives du formulaire
   873→    $steps = $pdo->prepare("
   874→        SELECT st.*, GROUP_CONCAT(sr.email, '|') as emails
   875→        FROM steps st
   876→        JOIN step_recipients sr ON sr.step_id = st.id
   877→        WHERE st.form_id = ? AND st.actif = 1
   878→        GROUP BY st.id
   879→        ORDER BY st.ordre ASC, st.id ASC
   880→    ");
   881→    $steps->execute([$submission['form_id']]);
   882→    $all_steps = $steps->fetchAll(PDO::FETCH_ASSOC);
   883→
   884→    if (empty($all_steps)) return;
   885→
   886→    // Tokens déjà générés pour cette soumission
   887→    $existing = $pdo->prepare("SELECT step_id, done_at FROM tokens WHERE submission_id = ?");
   888→    $existing->execute([$submission_id]);
   889→    $tokens_by_step = [];
   890→    foreach ($existing->fetchAll(PDO::FETCH_ASSOC) as $t) {
   891→        $tokens_by_step[$t['step_id']][] = $t['done_at'];
   892→    }
   893→
   894→    // Groupe les étapes par ordre
   895→    $by_ordre = [];
   896→    foreach ($all_steps as $step) {
   897→        $by_ordre[$step['ordre']][] = $step;
   898→    }
   899→    ksort($by_ordre);
   900→
   901→    foreach ($by_ordre as $ordre => $groupe) {
   902→        $step_ids    = array_column($groupe, 'id');
   903→        $all_started = count(array_intersect($step_ids, array_keys($tokens_by_step))) === count($step_ids);
   904→        $all_done    = $all_started && array_reduce($step_ids, function($carry, $sid) use ($tokens_by_step) {
   905→            if (!$carry) return false;
   906→            if (!isset($tokens_by_step[$sid])) return false;
   907→            foreach ($tokens_by_step[$sid] as $done) {
   908→                if (empty($done)) return false;
   909→            }
   910→            return true;
   911→        }, true);
   912→
   913→        if (!$all_started) {
   914→            // Cette étape n'a pas encore de tokens → on la démarre (parallèle dans le groupe)
   915→            $now = date('Y-m-d H:i:s');
   916→            $expire_days = (int)get_setting('token_expire_days', '30');
   917→            $expires_at = date('Y-m-d H:i:s', strtotime("+{$expire_days} days"));
   918→            foreach ($groupe as $step) {
   919→                $emails = explode('|', $step['emails']);
   920→                foreach ($emails as $email) {
   921→                    $token = generate_token();
   922→                    $pdo->prepare("INSERT INTO tokens (submission_id, step_id, email, token, sent_at, expires_at) VALUES (?,?,?,?,?,?)")
   923→                        ->execute([$submission_id, $step['id'], $email, $token, $now, $expires_at]);
   924→                    $subject = '[Action requise] ' . ($submission['form_label'] ?? '') . ' — ' . $step['label'];
   925→                    $mail_sent = send_mail($email, $subject, build_mail_html($submission, $step['label'], $token));
   926→                    if (!$mail_sent) {
   927→                        error_log("Workflow: mail failed for token $token to $email");
   928→                    }
   929→                }
   930→            }
   931→            return; // On attend que cette étape soit terminée avant de passer à la suivante
   932→        }
   933→
   934→        if (!$all_done) {
   935→            return; // Étape en cours, on attend
   936→        }
   937→
   938→        // Étape terminée → on continue la boucle vers l'ordre suivant
   939→    }
   940→
   941→    // Toutes les étapes sont terminées → on close et on notifie l'agent
   942→    $now = date('Y-m-d H:i:s');
   943→    $pdo->prepare("UPDATE submissions SET closed_at = ?, status = 'valide' WHERE id = ?")
   944→        ->execute([$now, $submission_id]);
   945→
   946→    // Notification de validation finale a l'agent
   947→    $agent_email = $submission['submitted_by'] ?? '';
   948→    if (!empty($agent_email) && filter_var($agent_email, FILTER_VALIDATE_EMAIL)) {
   949→        $subject = 'Demande validée — ' . ($submission['form_label'] ?? 'Workflow DREETS');
   950→        $body = '<!DOCTYPE html><html lang="fr"><head><meta charset="UTF-8"></head>
   951→<body style="font-family:Arial,sans-serif;max-width:600px;margin:auto;padding:20px;color:#222;">
   952→  <h2 style="color:#1a6b3c;">✓ Demande validée</h2>
   953→  <p>Votre demande <strong>' . h($submission['form_label'] ?? '') . '</strong> a été <strong>validée</strong> par l\'ensemble des validateurs.</p>
   954→  <p>Le processus de workflow est désormais terminé.</p>
   955→  <p style="font-size:12px;color:#999;margin-top:24px;">Workflow DREETS — ' . h(get_setting('smtp_from', SMTP_FROM)) . '</p>
   956→</body></html>';
   957→        send_mail($agent_email, $subject, $body);
   958→    }
   959→
   960→    app_log('workflow_complete', 'submission:' . $submission_id, 'Formulaire ' . ($submission['form_label'] ?? '') . ' validé', $agent_email);
   961→
   962→    // Webhook notification
   963→    send_webhook('workflow_complete', ['submission_id' => $submission_id, 'form_label' => $submission['form_label'] ?? '', 'submitted_by' => $submission['submitted_by'] ?? '']);
   964→}

   966→/**
   967→ * Appelé par validate.php quand un token est validé.
   968→ * Met à jour done_at puis avance le workflow.
   969→ */
   970→function validate_token(string $token, string $action = 'valider', string $comment = ''): array {
   971→    $pdo = get_pdo();
   972→
   973→    $row = $pdo->prepare("
   974→        SELECT t.*, st.label as step_label, s.form_id,
   975→               f.label as form_label, s.data, s.closed_at, s.status
   976→        FROM tokens t
   977→        JOIN steps st ON st.id = t.step_id
   978→        JOIN submissions s ON s.id = t.submission_id
   979→        JOIN forms f ON f.id = s.form_id
   980→        WHERE t.token = ?
   981→    ");
   982→    $row->execute([$token]);
   983→    $t = $row->fetch(PDO::FETCH_ASSOC);
   984→
   985→    if (!$t)             return ['status' => 'invalid'];
   986→    if ($t['done_at'])   return ['status' => 'already_done', 'data' => $t];
   987→    if ($t['closed_at']) return ['status' => 'closed',       'data' => $t];
   988→
   989→    // Vérifier si le token a expiré
   990→    if (!empty($t['expires_at']) && strtotime($t['expires_at']) < time()) {
   991→        return ['status' => 'expired', 'data' => $t];
   992→    }
   993→
   994→    // Récupérer les données actuelles
   995→    $data = json_decode($t['data'], true);
   996→    
   997→    // Ajouter la validation au tableau des validations
   998→    $validation = [
   999→        'step_label' => $t['step_label'],
  1000→        'email' => $t['email'],
  1001→        'action' => $action,
  1002→        'commentaire' => $comment,
  1003→        'date' => date('Y-m-d H:i:s')
  1004→    ];
  1005→    
  1006→    // Initialiser le tableau des validations s'il n'existe pas
  1007→    if (!isset($data['validations'])) {
  1008→        $data['validations'] = [];
  1009→    }
  1010→    
  1011→    // Ajouter la nouvelle validation
  1012→    $data['validations'][] = $validation;
  1013→    
  1014→    // Mettre à jour les données avec les validations
  1015→    $updated_data = json_encode($data);
  1016→    
  1017→    if ($action === 'refuser') {
  1018→        // Pour le refus : mettre à jour done_at et fermer la soumission avec status refuse
  1019→        $pdo->prepare("UPDATE tokens SET done_at = ? WHERE token = ?")
  1020→            ->execute([date('Y-m-d H:i:s'), $token]);
  1021→            
  1022→        // Fermer la soumission avec le statut refuse
  1023→        $pdo->prepare("UPDATE submissions SET closed_at = ?, status = 'refuse' WHERE id = ?")
  1024→            ->execute([date('Y-m-d H:i:s'), $t['submission_id']]);
  1025→
  1026→        // Notifier l'agent du refus
  1027→        $agent_email = $t['submitted_by'] ?? '';
  1028→        if (!empty($agent_email) && filter_var($agent_email, FILTER_VALIDATE_EMAIL)) {
  1029→            $refuse_subject = 'Demande refusée — ' . ($t['form_label'] ?? 'Workflow DREETS');
  1030→            $refuse_body = '<!DOCTYPE html><html lang="fr"><head><meta charset="UTF-8"></head>
  1031→<body style="font-family:Arial,sans-serif;max-width:600px;margin:auto;padding:20px;color:#222;">
  1032→  <h2 style="color:#c0392b;">Demande refusée</h2>
  1033→  <p>Votre demande <strong>' . h($t['form_label'] ?? '') . '</strong> a été refusée à l\'étape <strong>' . h($t['step_label']) . '</strong>.</p>
  1034→  ' . (!empty($comment) ? '<p><strong>Motif :</strong> ' . h($comment) . '</p>' : '') . '
  1035→  <p style="font-size:12px;color:#999;margin-top:24px;">Workflow DREETS — ' . h(get_setting('smtp_from', SMTP_FROM)) . '</p>
  1036→</body></html>';
  1037→            send_mail($agent_email, $refuse_subject, $refuse_body);
  1038→        }
  1039→    } else {
  1040→        // Pour la validation : comportement normal
  1041→        $pdo->prepare("UPDATE tokens SET done_at = ? WHERE token = ?")
  1042→            ->execute([date('Y-m-d H:i:s'), $token]);
  1043→
  1044→        advance_workflow($t['submission_id']);
  1045→    }
  1046→    
  1047→    // Mettre à jour les données de la soumission
  1048→    $pdo->prepare("UPDATE submissions SET data = ? WHERE id = ?")
  1049→        ->execute([$updated_data, $t['submission_id']]);
  1050→
  1051→    // Webhook notification
  1052→    send_webhook('token_validated', ['submission_id' => $t['submission_id'], 'step_label' => $t['step_label'], 'email' => $t['email'], 'action' => $action]);
  1053→
  1054→    $t['done_at'] = date('Y-m-d H:i:s');
  1055→    return ['status' => 'ok', 'data' => $t];
  1056→}

  1385→/**
  1386→ * Régénère un token expiré pour un validateur (admin uniquement)
  1387→ * Invalide l'ancien token et crée un nouveau avec une nouvelle date d'expiration
  1388→ *
  1389→ * @param int $old_token_id ID de l'ancien token
  1390→ * @return array ['success' => bool, 'message' => string]
  1391→ */
  1392→function regenerate_token(int $old_token_id): array {
  1393→    $pdo = get_pdo();
  1394→
  1395→    // Récupérer l'ancien token
  1396→    $stmt = $pdo->prepare("
  1397→        SELECT t.*, s.status as sub_status
  1398→        FROM tokens t
  1399→        JOIN submissions s ON s.id = t.submission_id
  1400→        WHERE t.id = ?
  1401→    ");
  1402→    $stmt->execute([$old_token_id]);
  1403→    $old = $stmt->fetch(PDO::FETCH_ASSOC);
  1404→
  1405→    if (!$old) {
  1406→        return ['success' => false, 'message' => 'Token introuvable.'];
  1407→    }
  1408→    if ($old['done_at']) {
  1409→        return ['success' => false, 'message' => 'Ce token a déjà été traité.'];
  1410→    }
  1411→    if ($old['sub_status'] !== 'en_cours') {
  1412→        return ['success' => false, 'message' => 'La soumission n\'est plus en cours.'];
  1413→    }
  1414→
  1415→    // Marquer l'ancien token comme traité (invalidé)
  1416→    $pdo->prepare("UPDATE tokens SET done_at = ? WHERE id = ?")
  1417→        ->execute([date('Y-m-d H:i:s'), $old_token_id]);
  1418→
  1419→    // Créer un nouveau token
  1420→    $new_token = generate_token();
  1421→    $expire_days = (int)get_setting('token_expire_days', '30');
  1422→    $expires_at = date('Y-m-d H:i:s', strtotime("+{$expire_days} days"));
  1423→    $now = date('Y-m-d H:i:s');
  1424→
  1425→    $pdo->prepare("INSERT INTO tokens (submission_id, step_id, email, token, sent_at, expires_at) VALUES (?, ?, ?, ?, ?, ?)")
  1426→        ->execute([$old['submission_id'], $old['step_id'], $old['email'], $new_token, $now, $expires_at]);
  1427→
  1428→    // Envoyer le nouveau lien par email
  1429→    $sub_stmt = $pdo->prepare("
  1430→        SELECT s.*, f.label as form_label FROM submissions s
  1431→        JOIN forms f ON f.id = s.form_id WHERE s.id = ?
  1432→    ");
  1433→    $sub_stmt->execute([$old['submission_id']]);
  1434→    $submission = $sub_stmt->fetch(PDO::FETCH_ASSOC);
  1435→
  1436→    $step_stmt = $pdo->prepare("SELECT label FROM steps WHERE id = ?");
  1437→    $step_stmt->execute([$old['step_id']]);
  1438→    $step = $step_stmt->fetch(PDO::FETCH_ASSOC);
  1439→
  1440→    if ($submission && $step) {
  1441→        $subject = '[Renvoi] ' . ($submission['form_label'] ?? '') . ' — ' . ($step['label'] ?? '');
  1442→        $mail_sent = send_mail($old['email'], $subject, build_mail_html($submission, $step['label'], $new_token));
  1443→    }
  1444→
  1445→    app_log('token_regenerate', 'token:' . $old_token_id, 'Token régénéré pour ' . $old['email'] . ', nouveau token créé');
  1446→
  1447→    return [
  1448→        'success' => true,
  1449→        'message' => 'Nouveau lien de validation envoyé à ' . $old['email'],
  1450→    ];
  1451→}

  1455→/**
  1456→ * Annule une soumission en cours

  1462→function cancel_submission(int $submission_id, string $cancelled_by = ''): array {
  1463→    $pdo = get_pdo();
  1464→
  1465→    $stmt = $pdo->prepare("SELECT s.*, f.label as form_label FROM submissions s JOIN forms f ON f.id = s.form_id WHERE s.id = ?");

  1468→
  1469→    if (!$submission) {
  1470→        return ['success' => false, 'message' => 'Soumission introuvable.'];

  1493→        'date' => $now,
  1494→    ];
  1495→    $pdo->prepare("UPDATE submissions SET data = ? WHERE id = ?")
  1496→        ->execute([json_encode($data, JSON_UNESCAPED_UNICODE), $submission_id]);
  1497→
  1498→    // Notifier l'agent
  1499→    $agent_email = $submission['submitted_by'] ?? '';
  1500→    if (!empty($agent_email) && filter_var($agent_email, FILTER_VALIDATE_EMAIL)) {
  1501→        $subject = 'Demande annulée — ' . ($submission['form_label'] ?? 'Workflow DREETS');
  1502→        $body = '<!DOCTYPE html><html lang="fr"><head><meta charset="UTF-8"></head>
  1503→<body style="font-family:Arial,sans-serif;max-width:600px;margin:auto;padding:20px;color:#222;">
  1504→  <h2 style="color:#b45309;">Demande annulée</h2>
  1505→  <p>Votre demande <strong>' . h($submission['form_label'] ?? '') . '</strong> a été annulée.</p>
  1506→  <p style="font-size:12px;color:#999;margin-top:24px;">Workflow DREETS</p>
  1507→</body></html>';
  1508→        send_mail($agent_email, $subject, $body);
  1509→    }
  1510→
  1511→    app_log('submission_cancel', 'submission:' . $submission_id, 'Soumission annulée', $cancelled_by);
  1512→
  1513→    // Webhook notification
  1514→    send_webhook('submission_cancelled', ['submission_id' => $submission_id, 'form_label' => $submission['form_label'] ?? '', 'cancelled_by' => $cancelled_by]);
  1515→
  1516→    return ['success' => true, 'message' => 'Soumission annulée avec succès.'];
  1517→}

  1632→/**
  1633→ * Gère l'upload d'un fichier pour une soumission
  1634→ *
  1635→ * @param array $file Le tableau $_FILES['field_name']
  1636→ * @param int $submission_id ID de la soumission
  1637→ * @param string $field_name Nom du champ
  1638→ * @return array ['success' => bool, 'message' => string, 'attachment_id' => int|null]
  1639→ */
  1640→function handle_file_upload(array $file, int $submission_id, string $field_name): array {
  1641→    // Vérifier les erreurs d'upload
  1642→    if ($file['error'] !== UPLOAD_ERR_OK) {
  1643→        $errors = [
  1644→            UPLOAD_ERR_INI_SIZE   => 'Le fichier dépasse la taille maximale autorisée par le serveur.',
  1645→            UPLOAD_ERR_FORM_SIZE  => 'Le fichier dépasse la taille maximale autorisée par le formulaire.',
  1646→            UPLOAD_ERR_PARTIAL    => 'Le fichier n\'a été que partiellement téléchargé.',
  1647→            UPLOAD_ERR_NO_FILE    => 'Aucun fichier n\'a été téléchargé.',
  1648→            UPLOAD_ERR_NO_TMP_DIR => 'Dossier temporaire manquant sur le serveur.',
  1649→            UPLOAD_ERR_CANT_WRITE => 'Erreur d\'écriture sur le serveur.',
  1650→        ];
  1651→        return ['success' => false, 'message' => $errors[$file['error']] ?? 'Erreur inconnue lors de l\'upload.', 'attachment_id' => null];
  1652→    }
  1653→
  1654→    // Vérifier la taille
  1655→    if ($file['size'] > get_max_file_size()) {
  1656→        return ['success' => false, 'message' => 'Le fichier dépasse la taille maximale autorisée (10 Mo).', 'attachment_id' => null];
  1657→    }
  1658→
  1659→    // Vérifier l'extension
  1660→    $ext = strtolower(pathinfo($file['name'], PATHINFO_EXTENSION));
  1661→    if (!in_array($ext, get_allowed_extensions())) {
  1662→        return ['success' => false, 'message' => 'Type de fichier non autorisé. Extensions acceptées : ' . implode(', ', get_allowed_extensions()) . '.', 'attachment_id' => null];
  1663→    }
  1664→
  1665→    // Vérifier le type MIME
  1666→    $finfo = finfo_open(FILEINFO_MIME_TYPE);
  1667→    $mime_type = finfo_file($finfo, $file['tmp_name']);
  1668→    finfo_close($finfo);
  1669→    if (!in_array($mime_type, get_allowed_mime_types())) {
  1670→        return ['success' => false, 'message' => 'Type MIME non autorisé : ' . $mime_type . '.', 'attachment_id' => null];

  1769→
  1770→    if (!$tok) {
  1771→        return ['success' => false, 'message' => 'Token introuvable.'];
  1772→    }
  1773→    if ($tok['done_at']) {
  1774→        return ['success' => false, 'message' => 'Ce token a déjà été traité.'];
  1775→    }
  1776→    if ($tok['sub_status'] !== 'en_cours') {
  1777→        return ['success' => false, 'message' => 'La soumission n\'est plus en cours.'];
  1778→    }
  1779→
  1780→    // Valider l'email du délégataire

  1784→    }
  1785→    if ($to_email === $tok['email']) {
  1786→        return ['success' => false, 'message' => 'Vous ne pouvez pas déléguer à vous-même.'];
  1787→    }
  1788→
  1789→    // Vérifier qu'un token n'existe pas déjà pour cet email sur cette étape
  1790→    $dup_check = $pdo->prepare("SELECT 1 FROM tokens WHERE submission_id = ? AND step_id = ? AND email = ? AND done_at IS NULL");
  1791→    $dup_check->execute([$tok['submission_id'], $tok['step_id'], $to_email]);
  1792→    if ($dup_check->fetch()) {
  1793→        return ['success' => false, 'message' => 'Un token de validation est déjà actif pour ' . $to_email . ' sur cette étape.'];
  1794→    }

  1796→    // Marquer l'ancien token comme traité (délégué)
