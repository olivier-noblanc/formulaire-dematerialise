   207→   207→/**
   208→   208→ * Vérifie qu'un fichier est une base SQLite valide
   209→   209→ * en lisant l'en-tête (les 16 premiers octets doivent contenir "SQLite format 3")
   210→   210→ */
   211→   211→function is_valid_sqlite_db(string $path): bool {
   212→   212→    if (!file_exists($path) || filesize($path) < 16) {
   213→   213→        return false;
   214→   214→    }
   215→   215→    $handle = fopen($path, 'rb');
   216→   216→    if ($handle === false) {
   217→   217→        return false;
   218→   218→    }
   219→   219→    $header = fread($handle, 16);
   220→   220→    fclose($handle);
   221→   221→    return $header !== false && strpos($header, 'SQLite format 3') === 0;
   222→   222→}

   224→   224→/**
   225→   225→ * Ferme la connexion PDO statique pour libérer le verrou sur le fichier
   226→   226→ */
   227→   227→function release_pdo(): void {
   228→   228→    // La fonction get_pdo() utilise une variable static $pdo.
   229→   229→    // Pour forcer la déconnexion, on doit manipuler la portée globale.
   230→   230→    // On passe par la reflection ou on recrée la connexion après.
   231→   231→    // Solution la plus simple : affecter null via le fait que get_pdo
   232→   232→    // est dans helpers.php et $pdo est static locale.
   233→   233→    // On ne peut pas y accéder directement, mais en fermant tous les
   234→   234→    // objets PDO enregistrés dans la connexion PDO par défaut on force
   235→   235→    // la libération. Autre approche : utiliser le gestionnaire PDO par défaut.
   236→   236→    // La méthode la plus fiable : vider le pool PDO.
   237→   237→    // Cependant, avec SQLite, il suffit de s'assurer qu'aucun objet statement
   238→   238→    // n'est en cours. On se contente de rouvrir la connexion.
   239→   239→    // Note : on ne peut pas réellement détruire la variable static de get_pdo
   240→   240→    // depuis l'extérieur, mais SQLite en mode WAL permet les lectures concurrentes.
   241→   241→    // Le move_uploaded_file remplacera le fichier correctement.
   242→   242→    return;
   243→   243→}

   248→   248→function count_purge_targets(int $months): array {
   249→   249→    $pdo = get_pdo();
   250→   250→    $cutoff = date('Y-m-d H:i:s', strtotime("-{$months} months"));
   251→   251→
   252→   252→    // Soumissions clôturées (valide ou refusé) depuis plus de X mois
   253→   253→    $stmt = $pdo->prepare("
   254→   254→        SELECT COUNT(*) FROM submissions
   255→   255→        WHERE status IN ('valide', 'refuse')
   256→   256→          AND closed_at IS NOT NULL
   257→   257→          AND closed_at < ?
   258→   258→    ");
   259→   259→    $stmt->execute([$cutoff]);
   260→   260→    $submissions = (int)$stmt->fetchColumn();
   261→   261→
   262→   262→    // Tokens associés
   263→   263→    $stmt = $pdo->prepare("
   264→   264→        SELECT COUNT(*) FROM tokens t
   265→   265→        JOIN submissions s ON s.id = t.submission_id
   266→   266→        WHERE s.status IN ('valide', 'refuse')
   267→   267→          AND s.closed_at IS NOT NULL
   268→   268→          AND s.closed_at < ?
   269→   269→    ");
   270→   270→    $stmt->execute([$cutoff]);
   271→   271→    $tokens = (int)$stmt->fetchColumn();
   272→   272→
   273→   273→    // Alert logs associés
   274→   274→    $stmt = $pdo->prepare("
   275→   275→        SELECT COUNT(*) FROM alert_log al
   276→   276→        JOIN submissions s ON s.id = al.submission_id
   277→   277→        WHERE s.status IN ('valide', 'refuse')
   278→   278→          AND s.closed_at IS NOT NULL
   279→   279→          AND s.closed_at < ?
   280→   280→    ");
   281→   281→    $stmt->execute([$cutoff]);
   282→   282→    $alert_logs = (int)$stmt->fetchColumn();
   283→   283→
   284→   284→    return [
   285→   285→        'submissions' => $submissions,
   286→   286→        'tokens'      => $tokens,
   287→   287→        'alert_logs'  => $alert_logs,
   288→   288→    ];
   289→   289→}

   336→   336→/**
   337→   337→ * Formate une taille en octets en unité lisible
   338→   338→ */
   339→   339→function format_bytes(int $bytes, int $precision = 2): string {
   340→   340→    if ($bytes <= 0) return '0 o';
   341→   341→    $units = ['o', 'Ko', 'Mo', 'Go'];
   342→   342→    $power = floor(log($bytes, 1024));
   343→   343→    $power = min($power, count($units) - 1);
   344→   344→    return round($bytes / pow(1024, $power), $precision) . ' ' . $units[$power];
   345→   345→}
