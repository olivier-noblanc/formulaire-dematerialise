   143→// ── LAZY CRON ────────────────────────────────────────────────
   144→// Pas de cron sur le serveur. Le premier utilisateur qui se connecte
   145→// dans la journée/heure déclenche les tâches planifiées.
   146→
   147→/**
   148→ * Execute les tâches planifiées si nécessaire.
   149→ * Appelé automatiquement à chaque connexion (depuis get_pdo).
   150→ * 
   151→ * Fréquences :
   152→ * - daily  : alert_check (1x/jour)
   153→ * - hourly : remind (1x/heure)
   154→ */
   155→function run_lazy_cron(): void {
   156→    static $running = false;
   157→    if ($running) return; // Éviter la récursion (get_pdo() appelle run_lazy_cron())
   158→    $running = true;
   159→
   160→    $pdo = get_pdo();
   161→    $now = time();
   162→    
   163→    $tasks = [
   164→        'remind'      => ['interval' => 3600,      'file' => __DIR__ . '/remind.php'],
   165→        'alert_check' => ['interval' => 86400,     'file' => __DIR__ . '/alert_check.php'],
   166→    ];
   167→    
   168→    foreach ($tasks as $key => $task) {
   169→        // Vérifier la dernière exécution
   170→        $stmt = $pdo->prepare("SELECT last_run FROM lazy_cron WHERE task_key = ?");
   171→        $stmt->execute([$key]);
   172→        $last_run = $stmt->fetchColumn();
   173→        
   174→        $should_run = false;
   175→        if ($last_run === false || $last_run === null) {
   176→            $should_run = true;
   177→        } else {
   178→            $last_ts = strtotime($last_run);
   179→            if (($now - $last_ts) >= $task['interval']) {
   180→                $should_run = true;
   181→            }
   182→        }
   183→        
   184→        if ($should_run) {
   185→            // Verrouillage : éviter les exécutions concurrentes
   186→            try {
   187→                $pdo->prepare("INSERT OR REPLACE INTO lazy_cron (task_key, last_run, run_count) VALUES (?, ?, COALESCE((SELECT run_count FROM lazy_cron WHERE task_key = ?), 0) + 1)")
   188→                    ->execute([$key, date('Y-m-d H:i:s', $now), $key]);
   189→            } catch (PDOException $e) {
   190→                continue; // Un autre processus est en cours
   191→            }
   192→            
   193→            // Exécuter la tâche dans un bloc try/catch pour ne pas casser la page
   194→            try {
   195→                // Les scripts remind.php et alert_check.php sont conçus pour être
   196→                // appelés en CLI. On les inclut directement avec output buffering.
   197→                ob_start();
   198→                require $task['file'];
   199→                ob_end_clean();
   200→            } catch (\Throwable $e) {
   201→                // Ne pas faire échouer la page utilisateur
   202→                error_log("Lazy cron error ({$key}): " . $e->getMessage());
   203→            }
   204→        }
   205→    }
   206→}

   256→/**
   257→ * Genere un UUID v4 (RFC 4122 compliant)
   258→ * Utilise random_bytes() pour la securite cryptographique
   259→ */
   260→function generate_uuid(): string {
   261→    $data = random_bytes(16);
   262→    $data[6] = chr(ord($data[6]) & 0x0f | 0x40); // version 4
   263→    $data[8] = chr(ord($data[8]) & 0x3f | 0x80); // variant RFC 4122
   264→    return vsprintf('%s%s-%s-%s-%s-%s%s%s', str_split(bin2hex($data), 4));
   265→}

  1433→/**
  1434→ * Genere automatiquement un field_name a partir d'un libelle
  1435→ * Ex: "Date de prise de poste" → "date_de_prise_de_poste"
  1436→ * Ex: "Type d'arrivée" → "type_arrivee"
  1437→ */
  1438→function generate_field_name(string $label): string {
  1439→    // Minuscules (fallback si mbstring absent)
  1440→    $name = function_exists('mb_strtolower') ? mb_strtolower($label, 'UTF-8') : strtolower($label);
  1441→    // Supprimer les accents
  1442→    if (function_exists('transliterator_transliterate')) {
  1443→        $transliterated = transliterator_transliterate('Any-Latin; Latin-ASCII; Lower()', $name);
  1444→        if ($transliterated !== false) {
  1445→            $name = $transliterated;
  1446→        }
  1447→    }
  1448→    // Fallback manuel si intl pas dispo ou a échoué
  1449→    $name = str_replace(
  1450→        ['à','â','ä','é','è','ê','ë','ï','î','ô','ö','ù','û','ü','ç','œ','æ','ÿ'],
  1451→        ['a','a','a','e','e','e','e','i','i','o','o','u','u','u','c','oe','ae','y'],
  1452→        $name
  1453→    );
  1454→    // Remplacer tout ce qui n'est pas alphanumérique par un underscore
  1455→    $name = preg_replace('/[^a-z0-9]+/', '_', $name);
  1456→    // Nettoyer les underscores en double et en bordure
  1457→    $name = trim($name, '_');
  1458→    $name = preg_replace('/_+/', '_', $name);
  1459→    return $name ?: 'champ';
  1460→}

  1468→ * Le slug n'est JAMAIS visible par l'utilisateur final — c'est un identifiant

  1471→function generate_slug(string $label, ?string $exclude_form_id = null): string {
  1472→    $base = generate_field_name($label);

  1474→
  1475→    $pdo = get_pdo();
  1476→    $slug = $base;
  1477→    $suffix = 2;
  1478→
  1479→    while (true) {
  1480→        $sql = "SELECT COUNT(*) FROM forms WHERE slug = ?";
  1481→        $params = [$slug];
  1482→        if ($exclude_form_id !== null) {
  1483→            $sql .= " AND id != ?";
  1484→            $params[] = $exclude_form_id;

  1487→        $stmt->execute($params);
  1488→        if ((int)$stmt->fetchColumn() === 0) {
  1489→            return $slug;
  1490→        }
  1491→        $slug = $base . '_' . $suffix;
  1492→        $suffix++;
  1493→    }
  1494→}

  1501→function parse_options_input(string $input): ?string {

  1749→ * Prérequis : extension PHP ldap activée, accès réseau vers le serveur AD.
  1750→ * Connexion en lecture seule (bind anonyme ou compte de service dédié).

  1752→ * @param string $email Adresse email à vérifier
  1753→ * @return array ['ok' => bool, 'method' => string, 'detail' => string]
  1754→ */
  1755→function verify_email_ldap(string $email): array {

  1757→    if (!function_exists('ldap_connect')) {
  1758→        return ['ok' => false, 'method' => 'ldap', 'detail' => 'Extension PHP ldap non disponible'];
  1759→    }
  1760→
  1761→    $host     = get_setting('ldap_host', '');
  1762→    $port     = (int)get_setting('ldap_port', '389');
  1763→    $base_dn  = get_setting('ldap_base_dn', '');

  1776→        return ['ok' => false, 'method' => 'ldap', 'detail' => 'Impossible de se connecter au serveur LDAP ' . $host];

  1778→
  1779→    ldap_set_option($conn, LDAP_OPT_PROTOCOL_VERSION, 3);

  1786→        $bind = @ldap_bind($conn, $bind_dn, $bind_pass);

  1801→    $search_filter = str_replace($email, ldap_escape($email, '', LDAP_ESCAPE_FILTER), $search_filter);

  1814→    if ($count > 0) {

  1823→ * Vérifie qu'une adresse email existe via une probe SMTP (RCPT TO).
  1824→ *
  1825→ * Ouvre une connexion SMTP au serveur configuré, envoie HELO/MAIL FROM/RCPT TO
  1826→ * et vérifie si le serveur accepte le destinataire. Se déconnecte proprement
  1827→ * sans envoyer de mail (QUIT avant DATA).
  1828→ *
  1829→ * ⚠️ Attention : certains serveurs SMTP acceptent tous les RCPT TO (catch-all).
  1830→ *    Cette méthode est un indicateur, pas une garantie absolue.
  1831→ *
  1832→ * @param string $email Adresse email à vérifier
  1833→ * @return array ['ok' => bool, 'method' => string, 'detail' => string]
  1834→ */
  1835→function verify_email_smtp(string $email): array {

  1837→    $smtp_port = (int)get_setting('smtp_port', (string)SMTP_PORT);
  1838→    $smtp_from = get_setting('smtp_from', SMTP_FROM);
  1839→    $smtp_secure = get_setting('smtp_secure', '');
  1840→
  1841→    if (empty($smtp_host)) {
  1842→        return ['ok' => false, 'method' => 'smtp', 'detail' => 'Aucun serveur SMTP configuré'];
  1843→    }
  1844→
  1845→    // Vérifier que l'extension sockets est disponible
  1846→    if (!function_exists('fsockopen')) {
  1847→        return ['ok' => false, 'method' => 'smtp', 'detail' => 'Extension PHP sockets non disponible'];
  1848→    }
  1849→
  1850→    // Connexion SMTP avec timeout
  1851→    $timeout = 10;
  1852→    $errno = 0;
  1853→    $errstr = '';
  1854→
  1855→    // Pour TLS, on se connecte d'abord en clair puis on fait STARTTLS
  1856→    $conn = @fsockopen($smtp_host, $smtp_port, $errno, $errstr, $timeout);
  1857→    if (!$conn) {
  1858→        return ['ok' => false, 'method' => 'smtp', 'detail' => "Impossible de se connecter à $smtp_host:$smtp_port ($errstr)"];
  1859→    }
  1860→
  1861→    stream_set_timeout($conn, $timeout);
  1862→
  1863→    // Fonction utilitaire pour lire une réponse SMTP
  1864→    $read_smtp = function() use ($conn): string {
  1865→        $response = '';
  1866→        while ($line = fgets($conn, 512)) {
  1867→            $response .= $line;
  1868→            // Les réponses multilignes ont un '-' après le code, la dernière ligne a un espace
  1869→            if (isset($line[3]) && $line[3] === ' ') break;
  1870→        }
  1871→        return $response;
  1872→    };
  1873→

  1875→    $send_smtp = function(string $cmd) use ($conn): void {
  1876→        fwrite($conn, $cmd . "\r\n");
  1877→    };

  1880→    $banner = $read_smtp();

  1882→        fclose($conn);
  1883→        return ['ok' => false, 'method' => 'smtp', 'detail' => 'Bannière SMTP invalide : ' . trim($banner)];
  1884→    }
  1885→
  1886→    // HELO
  1887→    $send_smtp('HELO ' . gethostname());
  1888→    $resp = $read_smtp();

  1890→        fclose($conn);
  1891→        return ['ok' => false, 'method' => 'smtp', 'detail' => 'HELO rejeté : ' . trim($resp)];
  1892→    }
  1893→
  1894→    // STARTTLS si configuré
  1895→    if ($smtp_secure === 'tls') {
  1896→        $send_smtp('STARTTLS');
  1897→        $resp = $read_smtp();

  1900→            return ['ok' => false, 'method' => 'smtp', 'detail' => 'STARTTLS rejeté : ' . trim($resp)];
  1901→        }
  1902→        if (!@stream_socket_enable_crypto($conn, true, STREAM_CRYPTO_METHOD_TLS_CLIENT)) {
  1903→            fclose($conn);

  1906→        // Retour au HELO après STARTTLS
  1907→        $send_smtp('EHLO ' . gethostname());

  1909→        if (!str_starts_with($resp, '250')) {
  1910→            fclose($conn);
  1911→            return ['ok' => false, 'method' => 'smtp', 'detail' => 'EHLO après STARTTLS rejeté : ' . trim($resp)];

  1913→    }
  1914→
  1915→    // MAIL FROM
  1916→    $send_smtp('MAIL FROM:<' . $smtp_from . '>');

  1920→        $read_smtp();

  1929→    // QUIT proprement
  1930→    $send_smtp('QUIT');
  1931→    $read_smtp();
  1932→    fclose($conn);
  1933→
  1934→    $code = substr($resp, 0, 3);
  1935→    if ($code === '250') {
  1936→        return ['ok' => true, 'method' => 'smtp', 'detail' => 'Adresse acceptée par le serveur SMTP'];
  1937→    }
  1938→

  1942→    }
  1943→
  1944→    return ['ok' => false, 'method' => 'smtp', 'detail' => 'Adresse rejetée par le serveur SMTP : ' . trim($resp)];
  1945→}
