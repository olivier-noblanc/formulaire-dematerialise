   134→// ── LAZY CRON ────────────────────────────────────────────────
   135→// Pas de cron sur le serveur. Le premier utilisateur qui se connecte
   136→// dans la journée/heure déclenche les tâches planifiées.
   137→
   138→/**
   139→ * Execute les tâches planifiées si nécessaire.
   140→ * Appelé automatiquement à chaque connexion (depuis get_pdo).
   141→ * 
   142→ * Fréquences :
   143→ * - daily  : alert_check (1x/jour)
   144→ * - hourly : remind (1x/heure)
   145→ */
   146→function run_lazy_cron(): void {
   147→    $pdo = get_pdo();
   148→    $now = time();
   149→    
   150→    $tasks = [
   151→        'remind'      => ['interval' => 3600,      'file' => __DIR__ . '/remind.php'],
   152→        'alert_check' => ['interval' => 86400,     'file' => __DIR__ . '/alert_check.php'],
   153→    ];
   154→    
   155→    foreach ($tasks as $key => $task) {
   156→        // Vérifier la dernière exécution
   157→        $stmt = $pdo->prepare("SELECT last_run FROM lazy_cron WHERE task_key = ?");
   158→        $stmt->execute([$key]);
   159→        $last_run = $stmt->fetchColumn();
   160→        
   161→        $should_run = false;
   162→        if ($last_run === false || $last_run === null) {
   163→            $should_run = true;
   164→        } else {
   165→            $last_ts = strtotime($last_run);
   166→            if (($now - $last_ts) >= $task['interval']) {
   167→                $should_run = true;
   168→            }
   169→        }
   170→        
   171→        if ($should_run) {
   172→            // Verrouillage : éviter les exécutions concurrentes
   173→            try {
   174→                $pdo->prepare("INSERT OR REPLACE INTO lazy_cron (task_key, last_run, run_count) VALUES (?, ?, COALESCE((SELECT run_count FROM lazy_cron WHERE task_key = ?), 0) + 1)")
   175→                    ->execute([$key, date('Y-m-d H:i:s', $now), $key]);
   176→            } catch (PDOException $e) {
   177→                continue; // Un autre processus est en cours
   178→            }
   179→            
   180→            // Exécuter la tâche dans un bloc try/catch pour ne pas casser la page
   181→            try {
   182→                // Les scripts remind.php et alert_check.php sont conçus pour être
   183→                // appelés en CLI. On les inclut directement avec output buffering.
   184→                ob_start();
   185→                require $task['file'];
   186→                ob_end_clean();
   187→            } catch (\Throwable $e) {
   188→                // Ne pas faire échouer la page utilisateur
   189→                error_log("Lazy cron error ({$key}): " . $e->getMessage());
   190→            }
   191→        }
   192→    }
   193→}

   243→/**
   244→ * Genere un UUID v4 (RFC 4122 compliant)
   245→ * Utilise random_bytes() pour la securite cryptographique
   246→ */
   247→function generate_uuid(): string {
   248→    $data = random_bytes(16);
   249→    $data[6] = chr(ord($data[6]) & 0x0f | 0x40); // version 4
   250→    $data[8] = chr(ord($data[8]) & 0x3f | 0x80); // variant RFC 4122
   251→    return vsprintf('%s%s-%s-%s-%s-%s%s%s', str_split(bin2hex($data), 4));
   252→}

  1048→/**
  1049→ * Genere automatiquement un field_name a partir d'un libelle
  1050→ * Ex: "Date de prise de poste" → "date_de_prise_de_poste"
  1051→ * Ex: "Type d'arrivée" → "type_arrivee"
  1052→ */
  1053→function generate_field_name(string $label): string {
  1054→    // Minuscules
  1055→    $name = mb_strtolower($label, 'UTF-8');
  1056→    // Supprimer les accents
  1057→    $name = transliterator_transliterate('Any-Latin; Latin-ASCII; Lower()', $name);
  1058→    if ($name === false) {
  1059→        // Fallback manuel si intl pas dispo
  1060→        $name = str_replace(
  1061→            ['à','â','ä','é','è','ê','ë','ï','î','ô','ö','ù','û','ü','ç','œ','æ','ÿ'],
  1062→            ['a','a','a','e','e','e','e','i','i','o','o','u','u','u','c','oe','ae','y'],
  1063→            $name
  1064→        );
  1065→    }
  1066→    // Remplacer tout ce qui n'est pas alphanumérique par un underscore
  1067→    $name = preg_replace('/[^a-z0-9]+/', '_', $name);
  1068→    // Nettoyer les underscores en double et en bordure
  1069→    $name = trim($name, '_');
  1070→    $name = preg_replace('/_+/', '_', $name);
  1071→    return $name ?: 'champ';
  1072→}

  1078→ */
  1079→function parse_options_input(string $input): ?string {
  1080→    $input = trim($input);

  1084→    $decoded = json_decode($input, true);

  1086→        return $input;

  1271→// ── MOTEUR WORKFLOW ───────────────────────────────────────────
  1272→
  1273→/**
  1274→ * Déclenche la prochaine étape d'une soumission.
  1275→ * Appelé à la création ET après chaque validation.
  1276→ *
  1277→ * Logique :
  1278→ *  - Récupère toutes les étapes du formulaire triées par ordre
  1279→ *  - Trouve le plus petit ordre sans tokens générés = prochaine étape
  1280→ *  - Si ordre précédent non terminé (séquentiel) → on attend
  1281→ *  - Génère les tokens pour tous les destinataires de l'étape courante
  1282→ *  - Si plus aucune étape → soumission close
  1283→ */
  1284→function advance_workflow(int $submission_id): void {
  1285→    $pdo = get_pdo();
  1286→
  1287→    $sub = $pdo->prepare("
  1288→        SELECT s.*, f.label as form_label
  1289→        FROM submissions s
  1290→        JOIN forms f ON f.id = s.form_id
  1291→        WHERE s.id = ?
  1292→    ");
  1293→    $sub->execute([$submission_id]);
  1294→    $submission = $sub->fetch(PDO::FETCH_ASSOC);
  1295→    if (!$submission || $submission['closed_at']) return;
  1296→
  1297→    // Toutes les étapes actives du formulaire
  1298→    $steps = $pdo->prepare("
  1299→        SELECT st.*, GROUP_CONCAT(sr.email, '|') as emails
  1300→        FROM steps st
  1301→        JOIN step_recipients sr ON sr.step_id = st.id
  1302→        WHERE st.form_id = ? AND st.actif = 1
  1303→        GROUP BY st.id
  1304→        ORDER BY st.ordre ASC, st.id ASC
  1305→    ");
  1306→    $steps->execute([$submission['form_id']]);
  1307→    $all_steps = $steps->fetchAll(PDO::FETCH_ASSOC);
  1308→
  1309→    if (empty($all_steps)) return;
  1310→
  1311→    // Tokens déjà générés pour cette soumission
  1312→    $existing = $pdo->prepare("SELECT step_id, done_at FROM tokens WHERE submission_id = ?");
  1313→    $existing->execute([$submission_id]);
  1314→    $tokens_by_step = [];
  1315→    foreach ($existing->fetchAll(PDO::FETCH_ASSOC) as $t) {
  1316→        $tokens_by_step[$t['step_id']][] = $t['done_at'];
  1317→    }
  1318→
  1319→    // Groupe les étapes par ordre
  1320→    $by_ordre = [];
  1321→    foreach ($all_steps as $step) {
  1322→        $by_ordre[$step['ordre']][] = $step;
  1323→    }
  1324→    ksort($by_ordre);
  1325→
  1326→    foreach ($by_ordre as $ordre => $groupe) {
  1327→        $step_ids    = array_column($groupe, 'id');
  1328→        $all_started = count(array_intersect($step_ids, array_keys($tokens_by_step))) === count($step_ids);
  1329→        $all_done    = $all_started && array_reduce($step_ids, function($carry, $sid) use ($tokens_by_step) {
  1330→            if (!$carry) return false;
  1331→            if (!isset($tokens_by_step[$sid])) return false;
  1332→            foreach ($tokens_by_step[$sid] as $done) {
  1333→                if (empty($done)) return false;
  1334→            }
  1335→            return true;
  1336→        }, true);
  1337→
  1338→        if (!$all_started) {
  1339→            // Cette étape n'a pas encore de tokens → on la démarre (parallèle dans le groupe)
  1340→            $now = date('Y-m-d H:i:s');
  1341→            $expire_days = (int)get_setting('token_expire_days', '30');
  1342→            $expires_at = date('Y-m-d H:i:s', strtotime("+{$expire_days} days"));
  1343→            foreach ($groupe as $step) {
  1344→                $emails = explode('|', $step['emails']);
  1345→                foreach ($emails as $email) {
  1346→                    $token = generate_token();
  1347→                    $pdo->prepare("INSERT INTO tokens (submission_id, step_id, email, token, sent_at, expires_at) VALUES (?,?,?,?,?,?)")
  1348→                        ->execute([$submission_id, $step['id'], $email, $token, $now, $expires_at]);
  1349→                    $subject = '[Action requise] ' . ($submission['form_label'] ?? '') . ' — ' . $step['label'];
  1350→                    $mail_sent = send_mail($email, $subject, build_mail_html($submission, $step['label'], $token));
  1351→                    if (!$mail_sent) {
  1352→                        error_log("Workflow: mail failed for token $token to $email");
  1353→                    }
  1354→                }
  1355→            }
  1356→            return; // On attend que cette étape soit terminée avant de passer à la suivante
  1357→        }
  1358→
  1359→        if (!$all_done) {
  1360→            return; // Étape en cours, on attend
  1361→        }
  1362→
  1363→        // Étape terminée → on continue la boucle vers l'ordre suivant
  1364→    }
  1365→
  1366→    // Toutes les étapes sont terminées → on close et on notifie l'agent
  1367→    $now = date('Y-m-d H:i:s');
  1368→    $pdo->prepare("UPDATE submissions SET closed_at = ?, status = 'valide' WHERE id = ?")
  1369→        ->execute([$now, $submission_id]);
  1370→
  1371→    // Notification de validation finale a l'agent
  1372→    $agent_email = $submission['submitted_by'] ?? '';
  1373→    if (!empty($agent_email) && filter_var($agent_email, FILTER_VALIDATE_EMAIL)) {
  1374→        $subject = 'Demande validée — ' . ($submission['form_label'] ?? 'Workflow DREETS');
  1375→        $body = '<!DOCTYPE html><html lang="fr"><head><meta charset="UTF-8"></head>
  1376→<body style="font-family:Arial,sans-serif;max-width:600px;margin:auto;padding:20px;color:#222;">
  1377→  <h2 style="color:#1a6b3c;">✓ Demande validée</h2>
  1378→  <p>Votre demande <strong>' . h($submission['form_label'] ?? '') . '</strong> a été <strong>validée</strong> par l\'ensemble des validateurs.</p>
  1379→  <p>Le processus de workflow est désormais terminé.</p>
  1380→  <p style="font-size:12px;color:#999;margin-top:24px;">Workflow DREETS — ' . h(get_setting('smtp_from', SMTP_FROM)) . '</p>
  1381→</body></html>';
  1382→        send_mail($agent_email, $subject, $body);
  1383→    }
  1384→
  1385→    app_log('workflow_complete', 'submission:' . $submission_id, 'Formulaire ' . ($submission['form_label'] ?? '') . ' validé', $agent_email);
  1386→
  1387→    // Webhook notification
  1388→    send_webhook('workflow_complete', ['submission_id' => $submission_id, 'form_label' => $submission['form_label'] ?? '', 'submitted_by' => $submission['submitted_by'] ?? '']);
  1389→}

  1391→/**
  1392→ * Appelé par validate.php quand un token est validé.
  1393→ * Met à jour done_at puis avance le workflow.
  1394→ */
  1395→function validate_token(string $token, string $action = 'valider', string $comment = ''): array {
  1396→    $pdo = get_pdo();
  1397→
  1398→    $row = $pdo->prepare("
  1399→        SELECT t.*, st.label as step_label, s.form_id,
  1400→               f.label as form_label, s.data, s.closed_at, s.status
  1401→        FROM tokens t
  1402→        JOIN steps st ON st.id = t.step_id
  1403→        JOIN submissions s ON s.id = t.submission_id
  1404→        JOIN forms f ON f.id = s.form_id
  1405→        WHERE t.token = ?
  1406→    ");
  1407→    $row->execute([$token]);
  1408→    $t = $row->fetch(PDO::FETCH_ASSOC);
  1409→
  1410→    if (!$t)             return ['status' => 'invalid'];
  1411→    if ($t['done_at'])   return ['status' => 'already_done', 'data' => $t];
  1412→    if ($t['closed_at']) return ['status' => 'closed',       'data' => $t];
  1413→
  1414→    // Vérifier si le token a expiré
  1415→    if (!empty($t['expires_at']) && strtotime($t['expires_at']) < time()) {
  1416→        return ['status' => 'expired', 'data' => $t];
  1417→    }
  1418→
  1419→    // Récupérer les données actuelles
  1420→    $data = json_decode($t['data'], true);
  1421→    
  1422→    // Ajouter la validation au tableau des validations
  1423→    $validation = [
  1424→        'step_label' => $t['step_label'],
  1425→        'email' => $t['email'],
  1426→        'action' => $action,
  1427→        'commentaire' => $comment,
  1428→        'date' => date('Y-m-d H:i:s')
  1429→    ];
  1430→    
  1431→    // Initialiser le tableau des validations s'il n'existe pas
  1432→    if (!isset($data['validations'])) {
  1433→        $data['validations'] = [];
  1434→    }
  1435→    
  1436→    // Ajouter la nouvelle validation
  1437→    $data['validations'][] = $validation;
  1438→    
  1439→    // Mettre à jour les données avec les validations
  1440→    $updated_data = json_encode($data);
  1441→    
  1442→    if ($action === 'refuser') {
  1443→        // Pour le refus : mettre à jour done_at et fermer la soumission avec status refuse
  1444→        $pdo->prepare("UPDATE tokens SET done_at = ? WHERE token = ?")
  1445→            ->execute([date('Y-m-d H:i:s'), $token]);
  1446→            
  1447→        // Fermer la soumission avec le statut refuse
  1448→        $pdo->prepare("UPDATE submissions SET closed_at = ?, status = 'refuse' WHERE id = ?")
  1449→            ->execute([date('Y-m-d H:i:s'), $t['submission_id']]);
  1450→
  1451→        // Notifier l'agent du refus
  1452→        $agent_email = $t['submitted_by'] ?? '';
  1453→        if (!empty($agent_email) && filter_var($agent_email, FILTER_VALIDATE_EMAIL)) {
  1454→            $refuse_subject = 'Demande refusée — ' . ($t['form_label'] ?? 'Workflow DREETS');
  1455→            $refuse_body = '<!DOCTYPE html><html lang="fr"><head><meta charset="UTF-8"></head>
  1456→<body style="font-family:Arial,sans-serif;max-width:600px;margin:auto;padding:20px;color:#222;">
  1457→  <h2 style="color:#c0392b;">Demande refusée</h2>
  1458→  <p>Votre demande <strong>' . h($t['form_label'] ?? '') . '</strong> a été refusée à l\'étape <strong>' . h($t['step_label']) . '</strong>.</p>
  1459→  ' . (!empty($comment) ? '<p><strong>Motif :</strong> ' . h($comment) . '</p>' : '') . '
  1460→  <p style="font-size:12px;color:#999;margin-top:24px;">Workflow DREETS — ' . h(get_setting('smtp_from', SMTP_FROM)) . '</p>
  1461→</body></html>';
  1462→            send_mail($agent_email, $refuse_subject, $refuse_body);
  1463→        }
  1464→    } else {
  1465→        // Pour la validation : comportement normal
  1466→        $pdo->prepare("UPDATE tokens SET done_at = ? WHERE token = ?")
  1467→            ->execute([date('Y-m-d H:i:s'), $token]);
  1468→
  1469→        advance_workflow($t['submission_id']);
  1470→    }
  1471→    
  1472→    // Mettre à jour les données de la soumission
  1473→    $pdo->prepare("UPDATE submissions SET data = ? WHERE id = ?")
  1474→        ->execute([$updated_data, $t['submission_id']]);
  1475→
  1476→    // Webhook notification
  1477→    send_webhook('token_validated', ['submission_id' => $t['submission_id'], 'step_label' => $t['step_label'], 'email' => $t['email'], 'action' => $action]);
  1478→
  1479→    $t['done_at'] = date('Y-m-d H:i:s');
  1480→    return ['status' => 'ok', 'data' => $t];
  1481→}

  1808→// ── TOKEN REGENERATION ───────────────────────────────────────
  1809→
  1810→/**
  1811→ * Régénère un token expiré pour un validateur (admin uniquement)
  1812→ * Invalide l'ancien token et crée un nouveau avec une nouvelle date d'expiration
  1813→ *
  1814→ * @param int $old_token_id ID de l'ancien token
  1815→ * @return array ['success' => bool, 'message' => string]
  1816→ */
  1817→function regenerate_token(int $old_token_id): array {
  1818→    $pdo = get_pdo();
  1819→
  1820→    // Récupérer l'ancien token
  1821→    $stmt = $pdo->prepare("
  1822→        SELECT t.*, s.status as sub_status
  1823→        FROM tokens t
  1824→        JOIN submissions s ON s.id = t.submission_id
  1825→        WHERE t.id = ?
  1826→    ");
  1827→    $stmt->execute([$old_token_id]);
  1828→    $old = $stmt->fetch(PDO::FETCH_ASSOC);
  1829→
  1830→    if (!$old) {
  1831→        return ['success' => false, 'message' => 'Token introuvable.'];
  1832→    }
  1833→    if ($old['done_at']) {
  1834→        return ['success' => false, 'message' => 'Ce token a déjà été traité.'];
  1835→    }
  1836→    if ($old['sub_status'] !== 'en_cours') {
  1837→        return ['success' => false, 'message' => 'La soumission n\'est plus en cours.'];
  1838→    }
  1839→
  1840→    // Marquer l'ancien token comme traité (invalidé)
  1841→    $pdo->prepare("UPDATE tokens SET done_at = ? WHERE id = ?")
  1842→        ->execute([date('Y-m-d H:i:s'), $old_token_id]);
  1843→
  1844→    // Créer un nouveau token
  1845→    $new_token = generate_token();
  1846→    $expire_days = (int)get_setting('token_expire_days', '30');
  1847→    $expires_at = date('Y-m-d H:i:s', strtotime("+{$expire_days} days"));
  1848→    $now = date('Y-m-d H:i:s');
  1849→
  1850→    $pdo->prepare("INSERT INTO tokens (submission_id, step_id, email, token, sent_at, expires_at) VALUES (?, ?, ?, ?, ?, ?)")
  1851→        ->execute([$old['submission_id'], $old['step_id'], $old['email'], $new_token, $now, $expires_at]);
  1852→
  1853→    // Envoyer le nouveau lien par email
  1854→    $sub_stmt = $pdo->prepare("
  1855→        SELECT s.*, f.label as form_label FROM submissions s
  1856→        JOIN forms f ON f.id = s.form_id WHERE s.id = ?
  1857→    ");
  1858→    $sub_stmt->execute([$old['submission_id']]);
  1859→    $submission = $sub_stmt->fetch(PDO::FETCH_ASSOC);
  1860→
  1861→    $step_stmt = $pdo->prepare("SELECT label FROM steps WHERE id = ?");
  1862→    $step_stmt->execute([$old['step_id']]);
  1863→    $step = $step_stmt->fetch(PDO::FETCH_ASSOC);
  1864→
