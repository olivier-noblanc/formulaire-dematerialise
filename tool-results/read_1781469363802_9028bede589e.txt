   134→// ── LAZY CRON ────────────────────────────────────────────────
   135→// Pas de cron sur le serveur. Le premier utilisateur qui se connecte
   136→// dans la journée/heure déclenche les tâches planifiées.
   137→
   138→/**
   139→ * Execute les tâches planifiées si nécessaire.
   140→ * Appelé automatiquement à chaque connexion (depuis get_pdo).
   141→ * 
   142→ * Fréquences :
   143→ * - daily  : alert_check (1x/jour)
   144→ * - hourly : remind (1x/heure)
   145→ */
   146→function run_lazy_cron(): void {
   147→    $pdo = get_pdo();
   148→    $now = time();
   149→    
   150→    $tasks = [
   151→        'remind'      => ['interval' => 3600,      'file' => __DIR__ . '/remind.php'],
   152→        'alert_check' => ['interval' => 86400,     'file' => __DIR__ . '/alert_check.php'],
   153→    ];
   154→    
   155→    foreach ($tasks as $key => $task) {
   156→        // Vérifier la dernière exécution
   157→        $stmt = $pdo->prepare("SELECT last_run FROM lazy_cron WHERE task_key = ?");
   158→        $stmt->execute([$key]);
   159→        $last_run = $stmt->fetchColumn();
   160→        
   161→        $should_run = false;
   162→        if ($last_run === false || $last_run === null) {
   163→            $should_run = true;
   164→        } else {
   165→            $last_ts = strtotime($last_run);
   166→            if (($now - $last_ts) >= $task['interval']) {
   167→                $should_run = true;
   168→            }
   169→        }
   170→        
   171→        if ($should_run) {
   172→            // Verrouillage : éviter les exécutions concurrentes
   173→            try {
   174→                $pdo->prepare("INSERT OR REPLACE INTO lazy_cron (task_key, last_run, run_count) VALUES (?, ?, COALESCE((SELECT run_count FROM lazy_cron WHERE task_key = ?), 0) + 1)")
   175→                    ->execute([$key, date('Y-m-d H:i:s', $now), $key]);
   176→            } catch (PDOException $e) {
   177→                continue; // Un autre processus est en cours
   178→            }
   179→            
   180→            // Exécuter la tâche dans un bloc try/catch pour ne pas casser la page
   181→            try {
   182→                // Les scripts remind.php et alert_check.php sont conçus pour être
   183→                // appelés en CLI. On les inclut directement avec output buffering.
   184→                ob_start();
   185→                require $task['file'];
   186→                ob_end_clean();
   187→            } catch (\Throwable $e) {
   188→                // Ne pas faire échouer la page utilisateur
   189→                error_log("Lazy cron error ({$key}): " . $e->getMessage());
   190→            }
   191→        }
   192→    }
   193→}

   243→/**
   244→ * Genere un UUID v4 (RFC 4122 compliant)
   245→ * Utilise random_bytes() pour la securite cryptographique
   246→ */
   247→function generate_uuid(): string {
   248→    $data = random_bytes(16);
   249→    $data[6] = chr(ord($data[6]) & 0x0f | 0x40); // version 4
   250→    $data[8] = chr(ord($data[8]) & 0x3f | 0x80); // variant RFC 4122
   251→    return vsprintf('%s%s-%s-%s-%s-%s%s%s', str_split(bin2hex($data), 4));
   252→}

  1407→/**
  1408→ * Genere automatiquement un field_name a partir d'un libelle
  1409→ * Ex: "Date de prise de poste" → "date_de_prise_de_poste"
  1410→ * Ex: "Type d'arrivée" → "type_arrivee"
  1411→ */
  1412→function generate_field_name(string $label): string {
  1413→    // Minuscules (fallback si mbstring absent)
  1414→    $name = function_exists('mb_strtolower') ? mb_strtolower($label, 'UTF-8') : strtolower($label);
  1415→    // Supprimer les accents
  1416→    if (function_exists('transliterator_transliterate')) {
  1417→        $transliterated = transliterator_transliterate('Any-Latin; Latin-ASCII; Lower()', $name);
  1418→        if ($transliterated !== false) {
  1419→            $name = $transliterated;
  1420→        }
  1421→    }
  1422→    // Fallback manuel si intl pas dispo ou a échoué
  1423→    $name = str_replace(
  1424→        ['à','â','ä','é','è','ê','ë','ï','î','ô','ö','ù','û','ü','ç','œ','æ','ÿ'],
  1425→        ['a','a','a','e','e','e','e','i','i','o','o','u','u','u','c','oe','ae','y'],
  1426→        $name
  1427→    );
  1428→    // Remplacer tout ce qui n'est pas alphanumérique par un underscore
  1429→    $name = preg_replace('/[^a-z0-9]+/', '_', $name);
  1430→    // Nettoyer les underscores en double et en bordure
  1431→    $name = trim($name, '_');
  1432→    $name = preg_replace('/_+/', '_', $name);
  1433→    return $name ?: 'champ';
  1434→}

  1436→/**
  1437→ * Génère automatiquement un slug unique à partir d'un libellé.
  1438→ * Ex: "Onboarding agent" → "onboarding_agent"
  1439→ * Ex: "Demande de congé" → "demande_de_conge"

  1441→ *
  1442→ * Le slug n'est JAMAIS visible par l'utilisateur final — c'est un identifiant
  1443→ * technique interne utilisé uniquement dans les URLs (form.php?f=onboarding).

  1445→function generate_slug(string $label, ?string $exclude_form_id = null): string {

  1447→    if (empty($base)) $base = 'formulaire';
  1448→
  1449→    $pdo = get_pdo();
  1450→    $slug = $base;
  1451→    $suffix = 2;
  1452→
  1453→    while (true) {
  1454→        $sql = "SELECT COUNT(*) FROM forms WHERE slug = ?";
  1455→        $params = [$slug];
  1456→        if ($exclude_form_id !== null) {
  1457→            $sql .= " AND id != ?";
  1458→            $params[] = $exclude_form_id;
  1459→        }
  1460→        $stmt = $pdo->prepare($sql);
  1461→        $stmt->execute($params);
  1462→        if ((int)$stmt->fetchColumn() === 0) {
  1463→            return $slug;
  1464→        }
  1465→        $slug = $base . '_' . $suffix;
  1466→        $suffix++;
  1467→    }
  1468→}

  1475→function parse_options_input(string $input): ?string {

  1792→ * Ouvre une connexion SMTP au serveur configuré, envoie HELO/MAIL FROM/RCPT TO

  1804→    $smtp_port = (int)get_setting('smtp_port', (string)SMTP_PORT);

  1806→    $smtp_secure = get_setting('smtp_secure', '');

  1808→    if (empty($smtp_host)) {
  1809→        return ['ok' => false, 'method' => 'smtp', 'detail' => 'Aucun serveur SMTP configuré'];

  1815→    }

  1817→    // Connexion SMTP avec timeout
  1818→    $timeout = 10;
  1819→    $errno = 0;
  1820→    $errstr = '';
  1821→

  1823→    $conn = @fsockopen($smtp_host, $smtp_port, $errno, $errstr, $timeout);
  1824→    if (!$conn) {
  1825→        return ['ok' => false, 'method' => 'smtp', 'detail' => "Impossible de se connecter à $smtp_host:$smtp_port ($errstr)"];
  1826→    }
  1827→
  1828→    stream_set_timeout($conn, $timeout);
  1829→
  1830→    // Fonction utilitaire pour lire une réponse SMTP
  1831→    $read_smtp = function() use ($conn): string {
  1832→        $response = '';
  1833→        while ($line = fgets($conn, 512)) {
  1834→            $response .= $line;
  1835→            // Les réponses multilignes ont un '-' après le code, la dernière ligne a un espace
  1836→            if (isset($line[3]) && $line[3] === ' ') break;
  1837→        }
  1838→        return $response;
  1839→    };
  1840→
  1841→    // Fonction utilitaire pour envoyer une commande SMTP
  1842→    $send_smtp = function(string $cmd) use ($conn): void {
  1843→        fwrite($conn, $cmd . "\r\n");
  1844→    };

  1846→    // Bannière de bienvenue
  1847→    $banner = $read_smtp();
  1848→    if (!str_starts_with($banner, '220')) {
  1849→        fclose($conn);
  1850→        return ['ok' => false, 'method' => 'smtp', 'detail' => 'Bannière SMTP invalide : ' . trim($banner)];
  1851→    }
  1852→
  1853→    // HELO
  1854→    $send_smtp('HELO ' . gethostname());
  1855→    $resp = $read_smtp();
  1856→    if (!str_starts_with($resp, '250')) {
  1857→        fclose($conn);
  1858→        return ['ok' => false, 'method' => 'smtp', 'detail' => 'HELO rejeté : ' . trim($resp)];
  1859→    }
  1860→

  1870→            fclose($conn);

  1872→        }
  1873→        // Retour au HELO après STARTTLS
  1874→        $send_smtp('EHLO ' . gethostname());
  1875→        $resp = $read_smtp();
  1876→        if (!str_starts_with($resp, '250')) {
  1877→            fclose($conn);
  1878→            return ['ok' => false, 'method' => 'smtp', 'detail' => 'EHLO après STARTTLS rejeté : ' . trim($resp)];
  1879→        }
  1880→    }
  1881→
  1882→    // MAIL FROM
  1883→    $send_smtp('MAIL FROM:<' . $smtp_from . '>');
  1884→    $resp = $read_smtp();
  1885→    if (!str_starts_with($resp, '250')) {
  1886→        $send_smtp('QUIT');
  1887→        $read_smtp();
  1888→        fclose($conn);
  1889→        return ['ok' => false, 'method' => 'smtp', 'detail' => 'MAIL FROM rejeté : ' . trim($resp)];
  1890→    }
  1891→
  1892→    // RCPT TO — la vérification clé
  1893→    $send_smtp('RCPT TO:<' . $email . '>');
  1894→    $resp = $read_smtp();
  1895→
  1896→    // QUIT proprement
  1897→    $send_smtp('QUIT');
  1898→    $read_smtp();
  1899→    fclose($conn);
  1900→
  1901→    $code = substr($resp, 0, 3);
  1902→    if ($code === '250') {
  1903→        return ['ok' => true, 'method' => 'smtp', 'detail' => 'Adresse acceptée par le serveur SMTP'];
  1904→    }
  1905→
  1906→    if ($code === '251') {
  1907→        // 251 = User not local; will forward to <forward-path>
  1908→        return ['ok' => true, 'method' => 'smtp', 'detail' => 'Adresse acceptée (transfert) par le serveur SMTP'];
  1909→    }
  1910→
  1911→    return ['ok' => false, 'method' => 'smtp', 'detail' => 'Adresse rejetée par le serveur SMTP : ' . trim($resp)];
  1912→}
