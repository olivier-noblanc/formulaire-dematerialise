   134→// ── LAZY CRON ────────────────────────────────────────────────
   135→// Pas de cron sur le serveur. Le premier utilisateur qui se connecte
   136→// dans la journée/heure déclenche les tâches planifiées.
   137→
   138→/**
   139→ * Execute les tâches planifiées si nécessaire.
   140→ * Appelé automatiquement à chaque connexion (depuis get_pdo).
   141→ * 
   142→ * Fréquences :
   143→ * - daily  : alert_check (1x/jour)
   144→ * - hourly : remind (1x/heure)
   145→ */
   146→function run_lazy_cron(): void {
   147→    $pdo = get_pdo();
   148→    $now = time();
   149→    
   150→    $tasks = [
   151→        'remind'      => ['interval' => 3600,      'file' => __DIR__ . '/remind.php'],
   152→        'alert_check' => ['interval' => 86400,     'file' => __DIR__ . '/alert_check.php'],
   153→    ];
   154→    
   155→    foreach ($tasks as $key => $task) {
   156→        // Vérifier la dernière exécution
   157→        $stmt = $pdo->prepare("SELECT last_run FROM lazy_cron WHERE task_key = ?");
   158→        $stmt->execute([$key]);
   159→        $last_run = $stmt->fetchColumn();
   160→        
   161→        $should_run = false;
   162→        if ($last_run === false || $last_run === null) {
   163→            $should_run = true;
   164→        } else {
   165→            $last_ts = strtotime($last_run);
   166→            if (($now - $last_ts) >= $task['interval']) {
   167→                $should_run = true;
   168→            }
   169→        }
   170→        
   171→        if ($should_run) {
   172→            // Verrouillage : éviter les exécutions concurrentes
   173→            try {
   174→                $pdo->prepare("INSERT OR REPLACE INTO lazy_cron (task_key, last_run, run_count) VALUES (?, ?, COALESCE((SELECT run_count FROM lazy_cron WHERE task_key = ?), 0) + 1)")
   175→                    ->execute([$key, date('Y-m-d H:i:s', $now), $key]);
   176→            } catch (PDOException $e) {
   177→                continue; // Un autre processus est en cours
   178→            }
   179→            
   180→            // Exécuter la tâche dans un bloc try/catch pour ne pas casser la page
   181→            try {
   182→                // Les scripts remind.php et alert_check.php sont conçus pour être
   183→                // appelés en CLI. On les inclut directement avec output buffering.
   184→                ob_start();
   185→                require $task['file'];
   186→                ob_end_clean();
   187→            } catch (\Throwable $e) {
   188→                // Ne pas faire échouer la page utilisateur
   189→                error_log("Lazy cron error ({$key}): " . $e->getMessage());
   190→            }
   191→        }
   192→    }
   193→}

   243→/**
   244→ * Genere un UUID v4 (RFC 4122 compliant)
   245→ * Utilise random_bytes() pour la securite cryptographique
   246→ */
   247→function generate_uuid(): string {
   248→    $data = random_bytes(16);
   249→    $data[6] = chr(ord($data[6]) & 0x0f | 0x40); // version 4
   250→    $data[8] = chr(ord($data[8]) & 0x3f | 0x80); // variant RFC 4122
   251→    return vsprintf('%s%s-%s-%s-%s-%s%s%s', str_split(bin2hex($data), 4));
   252→}

  1379→/**
  1380→ * Genere automatiquement un field_name a partir d'un libelle
  1381→ * Ex: "Date de prise de poste" → "date_de_prise_de_poste"
  1382→ * Ex: "Type d'arrivée" → "type_arrivee"
  1383→ */
  1384→function generate_field_name(string $label): string {
  1385→    // Minuscules (fallback si mbstring absent)
  1386→    $name = function_exists('mb_strtolower') ? mb_strtolower($label, 'UTF-8') : strtolower($label);
  1387→    // Supprimer les accents
  1388→    if (function_exists('transliterator_transliterate')) {
  1389→        $transliterated = transliterator_transliterate('Any-Latin; Latin-ASCII; Lower()', $name);
  1390→        if ($transliterated !== false) {
  1391→            $name = $transliterated;
  1392→        }
  1393→    }
  1394→    // Fallback manuel si intl pas dispo ou a échoué
  1395→    $name = str_replace(
  1396→        ['à','â','ä','é','è','ê','ë','ï','î','ô','ö','ù','û','ü','ç','œ','æ','ÿ'],

  1413→function parse_options_input(string $input): ?string {

  1605→// ── MOTEUR WORKFLOW ───────────────────────────────────────────
  1606→
  1607→/**
  1608→ * Déclenche la prochaine étape d'une soumission.
  1609→ * Appelé à la création ET après chaque validation.
  1610→ *
  1611→ * Logique :
  1612→ *  - Récupère toutes les étapes du formulaire triées par ordre
  1613→ *  - Trouve le plus petit ordre sans tokens générés = prochaine étape
  1614→ *  - Si ordre précédent non terminé (séquentiel) → on attend
  1615→ *  - Génère les tokens pour tous les destinataires de l'étape courante
  1616→ *  - Si plus aucune étape → soumission close
  1617→ */
  1618→function advance_workflow(string $submission_id): void {
  1619→    $pdo = get_pdo();
  1620→
  1621→    $sub = $pdo->prepare("
  1622→        SELECT s.*, f.label as form_label
  1623→        FROM submissions s
  1624→        JOIN forms f ON f.id = s.form_id
  1625→        WHERE s.id = ?
  1626→    ");
  1627→    $sub->execute([$submission_id]);
  1628→    $submission = $sub->fetch(PDO::FETCH_ASSOC);
  1629→    if (!$submission || $submission['closed_at']) return;
  1630→
  1631→    // Toutes les étapes actives du formulaire
  1632→    $steps = $pdo->prepare("
  1633→        SELECT st.*, GROUP_CONCAT(sr.email, '|') as emails
  1634→        FROM steps st
  1635→        JOIN step_recipients sr ON sr.step_id = st.id
  1636→        WHERE st.form_id = ? AND st.actif = 1
  1637→        GROUP BY st.id
  1638→        ORDER BY st.ordre ASC, st.id ASC
  1639→    ");
  1640→    $steps->execute([$submission['form_id']]);
  1641→    $all_steps = $steps->fetchAll(PDO::FETCH_ASSOC);
  1642→
  1643→    if (empty($all_steps)) return;
  1644→
  1645→    // Tokens déjà générés pour cette soumission
  1646→    $existing = $pdo->prepare("SELECT step_id, done_at FROM tokens WHERE submission_id = ?");
  1647→    $existing->execute([$submission_id]);
  1648→    $tokens_by_step = [];
  1649→    foreach ($existing->fetchAll(PDO::FETCH_ASSOC) as $t) {
  1650→        $tokens_by_step[$t['step_id']][] = $t['done_at'];
  1651→    }
  1652→
  1653→    // Groupe les étapes par ordre
  1654→    $by_ordre = [];
  1655→    foreach ($all_steps as $step) {
  1656→        $by_ordre[$step['ordre']][] = $step;
  1657→    }
  1658→    ksort($by_ordre);
  1659→
  1660→    foreach ($by_ordre as $ordre => $groupe) {
  1661→        $step_ids    = array_column($groupe, 'id');
  1662→        $all_started = count(array_intersect($step_ids, array_keys($tokens_by_step))) === count($step_ids);
  1663→        $all_done    = $all_started && array_reduce($step_ids, function($carry, $sid) use ($tokens_by_step) {
  1664→            if (!$carry) return false;
  1665→            if (!isset($tokens_by_step[$sid])) return false;
  1666→            foreach ($tokens_by_step[$sid] as $done) {
  1667→                if (empty($done)) return false;
  1668→            }
  1669→            return true;
  1670→        }, true);
  1671→
  1672→        if (!$all_started) {
  1673→            // Cette étape n'a pas encore de tokens → on la démarre (parallèle dans le groupe)
  1674→            $now = date('Y-m-d H:i:s');
  1675→            $expire_days = (int)get_setting('token_expire_days', '30');
  1676→            $expires_at = date('Y-m-d H:i:s', strtotime("+{$expire_days} days"));
  1677→            foreach ($groupe as $step) {
  1678→                $emails = explode('|', $step['emails']);
  1679→                foreach ($emails as $email) {
  1680→                    $token = generate_token();
  1681→                    $token_row_id = generate_uuid();
  1682→                    $pdo->prepare("INSERT INTO tokens (id, submission_id, step_id, email, token, sent_at, expires_at) VALUES (?,?,?,?,?,?,?)")
  1683→                        ->execute([$token_row_id, $submission_id, $step['id'], $email, $token, $now, $expires_at]);
  1684→                    $subject = '[Action requise] ' . ($submission['form_label'] ?? '') . ' — ' . $step['label'];
  1685→                    $mail_sent = send_mail($email, $subject, build_mail_html($submission, $step['label'], $token));
  1686→                    if (!$mail_sent) {
  1687→                        error_log("Workflow: mail failed for token $token to $email");
  1688→                    }
  1689→                }
  1690→            }
  1691→            return; // On attend que cette étape soit terminée avant de passer à la suivante
  1692→        }
  1693→
  1694→        if (!$all_done) {
  1695→            return; // Étape en cours, on attend
  1696→        }
  1697→
  1698→        // Étape terminée → on continue la boucle vers l'ordre suivant
  1699→    }
  1700→
  1701→    // Toutes les étapes sont terminées → on close et on notifie l'agent
  1702→    $now = date('Y-m-d H:i:s');
  1703→    $pdo->prepare("UPDATE submissions SET closed_at = ?, status = 'valide' WHERE id = ?")
  1704→        ->execute([$now, $submission_id]);
  1705→
  1706→    // Notification de validation finale a l'agent
  1707→    $agent_email = $submission['submitted_by'] ?? '';
  1708→    if (!empty($agent_email) && filter_var($agent_email, FILTER_VALIDATE_EMAIL)) {
  1709→        $subject = 'Demande validée — ' . ($submission['form_label'] ?? 'Workflow DREETS');
  1710→        $body = '<!DOCTYPE html><html lang="fr"><head><meta charset="UTF-8"></head>
  1711→<body style="font-family:Arial,sans-serif;max-width:600px;margin:auto;padding:20px;color:#222;">
  1712→  <h2 style="color:#1a6b3c;">✓ Demande validée</h2>
  1713→  <p>Votre demande <strong>' . h($submission['form_label'] ?? '') . '</strong> a été <strong>validée</strong> par l\'ensemble des validateurs.</p>
  1714→  <p>Le processus de workflow est désormais terminé.</p>
  1715→  <p style="font-size:12px;color:#999;margin-top:24px;">Workflow DREETS — ' . h(get_setting('smtp_from', SMTP_FROM)) . '</p>
  1716→</body></html>';
  1717→        send_mail($agent_email, $subject, $body);
  1718→    }
  1719→
  1720→    app_log('workflow_complete', 'submission:' . $submission_id, 'Formulaire ' . ($submission['form_label'] ?? '') . ' validé', $agent_email);
  1721→
  1722→    // Webhook notification
  1723→    send_webhook('workflow_complete', ['submission_id' => $submission_id, 'form_label' => $submission['form_label'] ?? '', 'submitted_by' => $submission['submitted_by'] ?? '']);
  1724→}

  1730→function validate_token(string $token, string $action = 'valider', string $comment = ''): array {
  1731→    $pdo = get_pdo();
  1732→
  1733→    $row = $pdo->prepare("
  1734→        SELECT t.*, st.label as step_label, s.form_id,
  1735→               f.label as form_label, s.data, s.closed_at, s.status
  1736→        FROM tokens t
  1737→        JOIN steps st ON st.id = t.step_id
  1738→        JOIN submissions s ON s.id = t.submission_id
  1739→        JOIN forms f ON f.id = s.form_id
  1740→        WHERE t.token = ?
  1741→    ");

  1743→    $t = $row->fetch(PDO::FETCH_ASSOC);
  1744→
  1745→    if (!$t)             return ['status' => 'invalid'];
  1746→    if ($t['done_at'])   return ['status' => 'already_done', 'data' => $t];
  1747→    if ($t['closed_at']) return ['status' => 'closed',       'data' => $t];
  1748→
  1749→    // Vérifier si le token a expiré
  1750→    if (!empty($t['expires_at']) && strtotime($t['expires_at']) < time()) {
  1751→        return ['status' => 'expired', 'data' => $t];
  1752→    }
  1753→
  1754→    // Récupérer les données actuelles
  1755→    $data = json_decode($t['data'], true);
  1756→    
  1757→    // Ajouter la validation au tableau des validations
  1758→    $validation = [
  1759→        'step_label' => $t['step_label'],
  1760→        'email' => $t['email'],
  1761→        'action' => $action,
  1762→        'commentaire' => $comment,
  1763→        'date' => date('Y-m-d H:i:s')
  1764→    ];
  1765→    
  1766→    // Initialiser le tableau des validations s'il n'existe pas
  1767→    if (!isset($data['validations'])) {
  1768→        $data['validations'] = [];
  1769→    }
  1770→    
  1771→    // Ajouter la nouvelle validation
  1772→    $data['validations'][] = $validation;
  1773→    
  1774→    // Mettre à jour les données avec les validations
  1775→    $updated_data = json_encode($data);
  1776→    
  1777→    if ($action === 'refuser') {
  1778→        // Pour le refus : mettre à jour done_at et fermer la soumission avec status refuse
  1779→        $pdo->prepare("UPDATE tokens SET done_at = ? WHERE token = ?")
  1780→            ->execute([date('Y-m-d H:i:s'), $token]);
  1781→

  1784→            ->execute([date('Y-m-d H:i:s'), $t['submission_id']]);

  1796→</body></html>';
  1797→            send_mail($agent_email, $refuse_subject, $refuse_body);
  1798→        }
  1799→    } else {
  1800→        // Pour la validation : comportement normal
  1801→        $pdo->prepare("UPDATE tokens SET done_at = ? WHERE token = ?")
  1802→            ->execute([date('Y-m-d H:i:s'), $token]);
  1803→
  1804→        advance_workflow($t['submission_id']);
  1805→    }
  1806→

  1808→    $pdo->prepare("UPDATE submissions SET data = ? WHERE id = ?")
  1809→        ->execute([$updated_data, $t['submission_id']]);
  1810→
  1811→    // Webhook notification

  1813→
  1814→    $t['done_at'] = date('Y-m-d H:i:s');
  1815→    return ['status' => 'ok', 'data' => $t];
  1816→}
