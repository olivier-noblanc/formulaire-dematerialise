   143→// ── LAZY CRON ────────────────────────────────────────────────
   144→// Pas de cron sur le serveur. Le premier utilisateur qui se connecte
   145→// dans la journée/heure déclenche les tâches planifiées.
   146→
   147→/**
   148→ * Execute les tâches planifiées si nécessaire.
   149→ * Appelé automatiquement à chaque connexion (depuis get_pdo).
   150→ * 
   151→ * Fréquences :
   152→ * - daily  : alert_check (1x/jour)
   153→ * - hourly : remind (1x/heure)
   154→ */
   155→function run_lazy_cron(): void {
   156→    static $running = false;
   157→    if ($running) return; // Éviter la récursion (get_pdo() appelle run_lazy_cron())
   158→    $running = true;
   159→
   160→    $pdo = get_pdo();
   161→    $now = time();
   162→    
   163→    $tasks = [
   164→        'remind'      => ['interval' => 3600,      'file' => __DIR__ . '/remind.php'],
   165→        'alert_check' => ['interval' => 86400,     'file' => __DIR__ . '/alert_check.php'],
   166→    ];
   167→    
   168→    foreach ($tasks as $key => $task) {
   169→        // Vérifier la dernière exécution
   170→        $stmt = $pdo->prepare("SELECT last_run FROM lazy_cron WHERE task_key = ?");
   171→        $stmt->execute([$key]);
   172→        $last_run = $stmt->fetchColumn();
   173→        
   174→        $should_run = false;
   175→        if ($last_run === false || $last_run === null) {
   176→            $should_run = true;
   177→        } else {
   178→            $last_ts = strtotime($last_run);
   179→            if (($now - $last_ts) >= $task['interval']) {
   180→                $should_run = true;
   181→            }
   182→        }
   183→        
   184→        if ($should_run) {
   185→            // Verrouillage : éviter les exécutions concurrentes
   186→            try {
   187→                $pdo->prepare("INSERT OR REPLACE INTO lazy_cron (task_key, last_run, run_count) VALUES (?, ?, COALESCE((SELECT run_count FROM lazy_cron WHERE task_key = ?), 0) + 1)")
   188→                    ->execute([$key, date('Y-m-d H:i:s', $now), $key]);
   189→            } catch (PDOException $e) {
   190→                continue; // Un autre processus est en cours
   191→            }
   192→            
   193→            // Exécuter la tâche dans un bloc try/catch pour ne pas casser la page
   194→            try {
   195→                // Les scripts remind.php et alert_check.php sont conçus pour être
   196→                // appelés en CLI. On les inclut directement avec output buffering.
   197→                ob_start();
   198→                require $task['file'];
   199→                ob_end_clean();
   200→            } catch (\Throwable $e) {
   201→                // Ne pas faire échouer la page utilisateur
   202→                error_log("Lazy cron error ({$key}): " . $e->getMessage());
   203→            }
   204→        }
   205→    }
   206→}

   256→/**
   257→ * Genere un UUID v4 (RFC 4122 compliant)
   258→ * Utilise random_bytes() pour la securite cryptographique
   259→ */
   260→function generate_uuid(): string {
   261→    $data = random_bytes(16);
   262→    $data[6] = chr(ord($data[6]) & 0x0f | 0x40); // version 4
   263→    $data[8] = chr(ord($data[8]) & 0x3f | 0x80); // variant RFC 4122
   264→    return vsprintf('%s%s-%s-%s-%s-%s%s%s', str_split(bin2hex($data), 4));
   265→}

  1513→/**
  1514→ * Genere automatiquement un field_name a partir d'un libelle
  1515→ * Ex: "Date de prise de poste" → "date_de_prise_de_poste"
  1516→ * Ex: "Type d'arrivée" → "type_arrivee"
  1517→ */
  1518→function generate_field_name(string $label): string {
  1519→    // Minuscules (fallback si mbstring absent)
  1520→    $name = function_exists('mb_strtolower') ? mb_strtolower($label, 'UTF-8') : strtolower($label);
  1521→    // Supprimer les accents
  1522→    if (function_exists('transliterator_transliterate')) {
  1523→        $transliterated = transliterator_transliterate('Any-Latin; Latin-ASCII; Lower()', $name);
  1524→        if ($transliterated !== false) {
  1525→            $name = $transliterated;
  1526→        }
  1527→    }
  1528→    // Fallback manuel si intl pas dispo ou a échoué
  1529→    $name = str_replace(
  1530→        ['à','â','ä','é','è','ê','ë','ï','î','ô','ö','ù','û','ü','ç','œ','æ','ÿ'],
  1531→        ['a','a','a','e','e','e','e','i','i','o','o','u','u','u','c','oe','ae','y'],
  1532→        $name
  1533→    );
  1534→    // Remplacer tout ce qui n'est pas alphanumérique par un underscore
  1535→    $name = preg_replace('/[^a-z0-9]+/', '_', $name);
  1536→    // Nettoyer les underscores en double et en bordure
  1537→    $name = trim($name, '_');
  1538→    $name = preg_replace('/_+/', '_', $name);
  1539→    return $name ?: 'champ';
  1540→}

  1542→/**
  1543→ * Génère automatiquement un slug unique à partir d'un libellé.
  1544→ * Ex: "Onboarding agent" → "onboarding_agent"
  1545→ * Ex: "Demande de congé" → "demande_de_conge"
  1546→ * Si le slug existe déjà, ajoute un suffixe numérique : "onboarding_agent_2"
  1547→ *
  1548→ * Le slug n'est JAMAIS visible par l'utilisateur final — c'est un identifiant
  1549→ * technique interne utilisé uniquement dans les URLs (form.php?f=onboarding).
  1550→ */
  1551→function generate_slug(string $label, ?string $exclude_form_id = null): string {
  1552→    $base = generate_field_name($label);
  1553→    if (empty($base)) $base = 'formulaire';
  1554→
  1555→    $pdo = get_pdo();
  1556→    $slug = $base;
  1557→    $suffix = 2;
  1558→
  1559→    while (true) {
  1560→        $sql = "SELECT COUNT(*) FROM forms WHERE slug = ?";
  1561→        $params = [$slug];
  1562→        if ($exclude_form_id !== null) {
  1563→            $sql .= " AND id != ?";
  1564→            $params[] = $exclude_form_id;
  1565→        }
  1566→        $stmt = $pdo->prepare($sql);
  1567→        $stmt->execute($params);
  1568→        if ((int)$stmt->fetchColumn() === 0) {
  1569→            return $slug;
  1570→        }
  1571→        $slug = $base . '_' . $suffix;
  1572→        $suffix++;
  1573→    }
  1574→}

  1576→/**
  1577→ * Convertit une liste d'options (une par ligne) en JSON array
  1578→ * Ex: "Option A\nOption B" → '["Option A","Option B"]'
  1579→ * Si c'est déjà du JSON valide, le retourne tel quel
  1580→ */
  1581→function parse_options_input(string $input): ?string {
  1582→    $input = trim($input);
  1583→    if (empty($input)) return null;
  1584→    
  1585→    // Vérifier si c'est déjà du JSON valide
  1586→    $decoded = json_decode($input, true);
  1587→    if (json_last_error() === JSON_ERROR_NONE && is_array($decoded)) {
  1588→        return $input;
  1589→    }
  1590→    
  1591→    // Traiter comme une liste une option par ligne
  1592→    $lines = array_filter(array_map('trim', explode("\n", $input)));
  1593→    if (empty($lines)) return null;
  1594→    
  1595→    return json_encode($lines, JSON_UNESCAPED_UNICODE);
  1596→}
