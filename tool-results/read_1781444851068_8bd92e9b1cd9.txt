   105→function generate_csrf_token(): string {
   106→    if (empty($_SESSION['csrf_token'])) {
   107→        $_SESSION['csrf_token'] = bin2hex(random_bytes(32));
   108→    }
   109→    return $_SESSION['csrf_token'];
   110→}
   111→
   112→function csrf_field(): string {
   113→    return '<input type="hidden" name="csrf_token" value="' . h(generate_csrf_token()) . '">';
   114→}

   116→function verify_csrf(): bool {
   117→    // Mode test : bypass CSRF
   118→    if (TEST_MODE) return true;
   119→    $token = $_POST['csrf_token'] ?? '';
   120→    return !empty($token) && hash_equals($_SESSION['csrf_token'] ?? '', $token);
   121→}

   837→/**
   838→ * Genere automatiquement un field_name a partir d'un libelle
   839→ * Ex: "Date de prise de poste" → "date_de_prise_de_poste"
   840→ * Ex: "Type d'arrivée" → "type_arrivee"
   841→ */
   842→function generate_field_name(string $label): string {
   843→    // Minuscules
   844→    $name = mb_strtolower($label, 'UTF-8');
   845→    // Supprimer les accents
   846→    $name = transliterator_transliterate('Any-Latin; Latin-ASCII; Lower()', $name);
   847→    if ($name === false) {
   848→        // Fallback manuel si intl pas dispo
   849→        $name = str_replace(
   850→            ['à','â','ä','é','è','ê','ë','ï','î','ô','ö','ù','û','ü','ç','œ','æ','ÿ'],
   851→            ['a','a','a','e','e','e','e','i','i','o','o','u','u','u','c','oe','ae','y'],
   852→            $name
   853→        );
   854→    }
   855→    // Remplacer tout ce qui n'est pas alphanumérique par un underscore
   856→    $name = preg_replace('/[^a-z0-9]+/', '_', $name);
   857→    // Nettoyer les underscores en double et en bordure
   858→    $name = trim($name, '_');
   859→    $name = preg_replace('/_+/', '_', $name);
   860→    return $name ?: 'champ';
   861→}

   863→/**
   864→ * Convertit une liste d'options (une par ligne) en JSON array
   865→ * Ex: "Option A\nOption B" → '["Option A","Option B"]'
   866→ * Si c'est déjà du JSON valide, le retourne tel quel
   867→ */
   868→function parse_options_input(string $input): ?string {
   869→    $input = trim($input);
   870→    if (empty($input)) return null;
   871→    
   872→    // Vérifier si c'est déjà du JSON valide
   873→    $decoded = json_decode($input, true);
   874→    if (json_last_error() === JSON_ERROR_NONE && is_array($decoded)) {
   875→        return $input;
   876→    }
   877→    
   878→    // Traiter comme une liste une option par ligne
   879→    $lines = array_filter(array_map('trim', explode("\n", $input)));
   880→    if (empty($lines)) return null;
   881→    
   882→    return json_encode($lines, JSON_UNESCAPED_UNICODE);
   883→}

   975→// ── MOTEUR WORKFLOW ───────────────────────────────────────────

   977→/**
   978→ * Déclenche la prochaine étape d'une soumission.
   979→ * Appelé à la création ET après chaque validation.
   980→ *
   981→ * Logique :
   982→ *  - Récupère toutes les étapes du formulaire triées par ordre
   983→ *  - Trouve le plus petit ordre sans tokens générés = prochaine étape
   984→ *  - Si ordre précédent non terminé (séquentiel) → on attend
   985→ *  - Génère les tokens pour tous les destinataires de l'étape courante
   986→ *  - Si plus aucune étape → soumission close
   987→ */
   988→function advance_workflow(int $submission_id): void {
   989→    $pdo = get_pdo();
   990→
   991→    $sub = $pdo->prepare("
   992→        SELECT s.*, f.label as form_label
   993→        FROM submissions s
   994→        JOIN forms f ON f.id = s.form_id
   995→        WHERE s.id = ?
   996→    ");
   997→    $sub->execute([$submission_id]);
   998→    $submission = $sub->fetch(PDO::FETCH_ASSOC);
   999→    if (!$submission || $submission['closed_at']) return;
  1000→
  1001→    // Toutes les étapes actives du formulaire
  1002→    $steps = $pdo->prepare("
  1003→        SELECT st.*, GROUP_CONCAT(sr.email, '|') as emails
  1004→        FROM steps st
  1005→        JOIN step_recipients sr ON sr.step_id = st.id
  1006→        WHERE st.form_id = ? AND st.actif = 1
  1007→        GROUP BY st.id
  1008→        ORDER BY st.ordre ASC, st.id ASC
  1009→    ");
  1010→    $steps->execute([$submission['form_id']]);
  1011→    $all_steps = $steps->fetchAll(PDO::FETCH_ASSOC);
  1012→
  1013→    if (empty($all_steps)) return;
  1014→
  1015→    // Tokens déjà générés pour cette soumission
  1016→    $existing = $pdo->prepare("SELECT step_id, done_at FROM tokens WHERE submission_id = ?");
  1017→    $existing->execute([$submission_id]);
  1018→    $tokens_by_step = [];
  1019→    foreach ($existing->fetchAll(PDO::FETCH_ASSOC) as $t) {
  1020→        $tokens_by_step[$t['step_id']][] = $t['done_at'];
  1021→    }
  1022→
  1023→    // Groupe les étapes par ordre
  1024→    $by_ordre = [];
  1025→    foreach ($all_steps as $step) {
  1026→        $by_ordre[$step['ordre']][] = $step;
  1027→    }
  1028→    ksort($by_ordre);
  1029→
  1030→    foreach ($by_ordre as $ordre => $groupe) {
  1031→        $step_ids    = array_column($groupe, 'id');
  1032→        $all_started = count(array_intersect($step_ids, array_keys($tokens_by_step))) === count($step_ids);
  1033→        $all_done    = $all_started && array_reduce($step_ids, function($carry, $sid) use ($tokens_by_step) {
  1034→            if (!$carry) return false;
  1035→            if (!isset($tokens_by_step[$sid])) return false;
  1036→            foreach ($tokens_by_step[$sid] as $done) {
  1037→                if (empty($done)) return false;
  1038→            }
  1039→            return true;
  1040→        }, true);
  1041→
  1042→        if (!$all_started) {
  1043→            // Cette étape n'a pas encore de tokens → on la démarre (parallèle dans le groupe)
  1044→            $now = date('Y-m-d H:i:s');
  1045→            $expire_days = (int)get_setting('token_expire_days', '30');
  1046→            $expires_at = date('Y-m-d H:i:s', strtotime("+{$expire_days} days"));
  1047→            foreach ($groupe as $step) {
  1048→                $emails = explode('|', $step['emails']);
  1049→                foreach ($emails as $email) {
  1050→                    $token = generate_token();
  1051→                    $pdo->prepare("INSERT INTO tokens (submission_id, step_id, email, token, sent_at, expires_at) VALUES (?,?,?,?,?,?)")
  1052→                        ->execute([$submission_id, $step['id'], $email, $token, $now, $expires_at]);
  1053→                    $subject = '[Action requise] ' . ($submission['form_label'] ?? '') . ' — ' . $step['label'];
  1054→                    $mail_sent = send_mail($email, $subject, build_mail_html($submission, $step['label'], $token));
  1055→                    if (!$mail_sent) {
  1056→                        error_log("Workflow: mail failed for token $token to $email");
  1057→                    }
  1058→                }
  1059→            }
  1060→            return; // On attend que cette étape soit terminée avant de passer à la suivante
  1061→        }
  1062→
  1063→        if (!$all_done) {
  1064→            return; // Étape en cours, on attend
  1065→        }
  1066→
  1067→        // Étape terminée → on continue la boucle vers l'ordre suivant
  1068→    }
  1069→
  1070→    // Toutes les étapes sont terminées → on close et on notifie l'agent
  1071→    $now = date('Y-m-d H:i:s');
  1072→    $pdo->prepare("UPDATE submissions SET closed_at = ?, status = 'valide' WHERE id = ?")
  1073→        ->execute([$now, $submission_id]);
  1074→
  1075→    // Notification de validation finale a l'agent
  1076→    $agent_email = $submission['submitted_by'] ?? '';
  1077→    if (!empty($agent_email) && filter_var($agent_email, FILTER_VALIDATE_EMAIL)) {
  1078→        $subject = 'Demande validée — ' . ($submission['form_label'] ?? 'Workflow DREETS');
  1079→        $body = '<!DOCTYPE html><html lang="fr"><head><meta charset="UTF-8"></head>
  1080→<body style="font-family:Arial,sans-serif;max-width:600px;margin:auto;padding:20px;color:#222;">
  1081→  <h2 style="color:#1a6b3c;">✓ Demande validée</h2>
  1082→  <p>Votre demande <strong>' . h($submission['form_label'] ?? '') . '</strong> a été <strong>validée</strong> par l\'ensemble des validateurs.</p>
  1083→  <p>Le processus de workflow est désormais terminé.</p>
  1084→  <p style="font-size:12px;color:#999;margin-top:24px;">Workflow DREETS — ' . h(get_setting('smtp_from', SMTP_FROM)) . '</p>
  1085→</body></html>';
  1086→        send_mail($agent_email, $subject, $body);
  1087→    }
  1088→
  1089→    app_log('workflow_complete', 'submission:' . $submission_id, 'Formulaire ' . ($submission['form_label'] ?? '') . ' validé', $agent_email);
  1090→
  1091→    // Webhook notification
  1092→    send_webhook('workflow_complete', ['submission_id' => $submission_id, 'form_label' => $submission['form_label'] ?? '', 'submitted_by' => $submission['submitted_by'] ?? '']);
  1093→}

  1095→/**
  1096→ * Appelé par validate.php quand un token est validé.
  1097→ * Met à jour done_at puis avance le workflow.
  1098→ */
  1099→function validate_token(string $token, string $action = 'valider', string $comment = ''): array {
  1100→    $pdo = get_pdo();
  1101→
  1102→    $row = $pdo->prepare("
  1103→        SELECT t.*, st.label as step_label, s.form_id,
  1104→               f.label as form_label, s.data, s.closed_at, s.status
  1105→        FROM tokens t
  1106→        JOIN steps st ON st.id = t.step_id
  1107→        JOIN submissions s ON s.id = t.submission_id
  1108→        JOIN forms f ON f.id = s.form_id
  1109→        WHERE t.token = ?
  1110→    ");
  1111→    $row->execute([$token]);
  1112→    $t = $row->fetch(PDO::FETCH_ASSOC);
  1113→
  1114→    if (!$t)             return ['status' => 'invalid'];
  1115→    if ($t['done_at'])   return ['status' => 'already_done', 'data' => $t];
  1116→    if ($t['closed_at']) return ['status' => 'closed',       'data' => $t];
  1117→
  1118→    // Vérifier si le token a expiré
  1119→    if (!empty($t['expires_at']) && strtotime($t['expires_at']) < time()) {
  1120→        return ['status' => 'expired', 'data' => $t];
  1121→    }
  1122→
  1123→    // Récupérer les données actuelles
  1124→    $data = json_decode($t['data'], true);
  1125→    
  1126→    // Ajouter la validation au tableau des validations
  1127→    $validation = [
  1128→        'step_label' => $t['step_label'],
  1129→        'email' => $t['email'],
  1130→        'action' => $action,
  1131→        'commentaire' => $comment,
  1132→        'date' => date('Y-m-d H:i:s')
  1133→    ];
  1134→    
  1135→    // Initialiser le tableau des validations s'il n'existe pas
  1136→    if (!isset($data['validations'])) {
  1137→        $data['validations'] = [];
  1138→    }
  1139→    
  1140→    // Ajouter la nouvelle validation
  1141→    $data['validations'][] = $validation;
  1142→    
  1143→    // Mettre à jour les données avec les validations
  1144→    $updated_data = json_encode($data);
  1145→    
  1146→    if ($action === 'refuser') {
  1147→        // Pour le refus : mettre à jour done_at et fermer la soumission avec status refuse
  1148→        $pdo->prepare("UPDATE tokens SET done_at = ? WHERE token = ?")
  1149→            ->execute([date('Y-m-d H:i:s'), $token]);
  1150→            
  1151→        // Fermer la soumission avec le statut refuse
  1152→        $pdo->prepare("UPDATE submissions SET closed_at = ?, status = 'refuse' WHERE id = ?")
  1153→            ->execute([date('Y-m-d H:i:s'), $t['submission_id']]);
  1154→
  1155→        // Notifier l'agent du refus
  1156→        $agent_email = $t['submitted_by'] ?? '';
  1157→        if (!empty($agent_email) && filter_var($agent_email, FILTER_VALIDATE_EMAIL)) {
  1158→            $refuse_subject = 'Demande refusée — ' . ($t['form_label'] ?? 'Workflow DREETS');
  1159→            $refuse_body = '<!DOCTYPE html><html lang="fr"><head><meta charset="UTF-8"></head>
  1160→<body style="font-family:Arial,sans-serif;max-width:600px;margin:auto;padding:20px;color:#222;">
  1161→  <h2 style="color:#c0392b;">Demande refusée</h2>
  1162→  <p>Votre demande <strong>' . h($t['form_label'] ?? '') . '</strong> a été refusée à l\'étape <strong>' . h($t['step_label']) . '</strong>.</p>
  1163→  ' . (!empty($comment) ? '<p><strong>Motif :</strong> ' . h($comment) . '</p>' : '') . '
  1164→  <p style="font-size:12px;color:#999;margin-top:24px;">Workflow DREETS — ' . h(get_setting('smtp_from', SMTP_FROM)) . '</p>
  1165→</body></html>';
  1166→            send_mail($agent_email, $refuse_subject, $refuse_body);
  1167→        }
  1168→    } else {
  1169→        // Pour la validation : comportement normal
  1170→        $pdo->prepare("UPDATE tokens SET done_at = ? WHERE token = ?")
  1171→            ->execute([date('Y-m-d H:i:s'), $token]);
  1172→
  1173→        advance_workflow($t['submission_id']);
  1174→    }
  1175→    
  1176→    // Mettre à jour les données de la soumission
  1177→    $pdo->prepare("UPDATE submissions SET data = ? WHERE id = ?")
  1178→        ->execute([$updated_data, $t['submission_id']]);
  1179→
  1180→    // Webhook notification
  1181→    send_webhook('token_validated', ['submission_id' => $t['submission_id'], 'step_label' => $t['step_label'], 'email' => $t['email'], 'action' => $action]);
  1182→
  1183→    $t['done_at'] = date('Y-m-d H:i:s');
  1184→    return ['status' => 'ok', 'data' => $t];
  1185→}

  1512→// ── TOKEN REGENERATION ───────────────────────────────────────
  1513→
  1514→/**
  1515→ * Régénère un token expiré pour un validateur (admin uniquement)
  1516→ * Invalide l'ancien token et crée un nouveau avec une nouvelle date d'expiration
  1517→ *
  1518→ * @param int $old_token_id ID de l'ancien token
  1519→ * @return array ['success' => bool, 'message' => string]
  1520→ */
  1521→function regenerate_token(int $old_token_id): array {
  1522→    $pdo = get_pdo();
  1523→
  1524→    // Récupérer l'ancien token
  1525→    $stmt = $pdo->prepare("
  1526→        SELECT t.*, s.status as sub_status
  1527→        FROM tokens t
  1528→        JOIN submissions s ON s.id = t.submission_id
  1529→        WHERE t.id = ?
  1530→    ");
  1531→    $stmt->execute([$old_token_id]);
  1532→    $old = $stmt->fetch(PDO::FETCH_ASSOC);
  1533→
  1534→    if (!$old) {
  1535→        return ['success' => false, 'message' => 'Token introuvable.'];
  1536→    }
  1537→    if ($old['done_at']) {
  1538→        return ['success' => false, 'message' => 'Ce token a déjà été traité.'];
  1539→    }
  1540→    if ($old['sub_status'] !== 'en_cours') {
  1541→        return ['success' => false, 'message' => 'La soumission n\'est plus en cours.'];
  1542→    }
  1543→
  1544→    // Marquer l'ancien token comme traité (invalidé)
  1545→    $pdo->prepare("UPDATE tokens SET done_at = ? WHERE id = ?")
  1546→        ->execute([date('Y-m-d H:i:s'), $old_token_id]);
  1547→
  1548→    // Créer un nouveau token
  1549→    $new_token = generate_token();
  1550→    $expire_days = (int)get_setting('token_expire_days', '30');
  1551→    $expires_at = date('Y-m-d H:i:s', strtotime("+{$expire_days} days"));
  1552→    $now = date('Y-m-d H:i:s');
  1553→
  1554→    $pdo->prepare("INSERT INTO tokens (submission_id, step_id, email, token, sent_at, expires_at) VALUES (?, ?, ?, ?, ?, ?)")
  1555→        ->execute([$old['submission_id'], $old['step_id'], $old['email'], $new_token, $now, $expires_at]);
  1556→
  1557→    // Envoyer le nouveau lien par email
  1558→    $sub_stmt = $pdo->prepare("
  1559→        SELECT s.*, f.label as form_label FROM submissions s
  1560→        JOIN forms f ON f.id = s.form_id WHERE s.id = ?
  1561→    ");
  1562→    $sub_stmt->execute([$old['submission_id']]);
  1563→    $submission = $sub_stmt->fetch(PDO::FETCH_ASSOC);
  1564→
  1565→    $step_stmt = $pdo->prepare("SELECT label FROM steps WHERE id = ?");
  1566→    $step_stmt->execute([$old['step_id']]);
  1567→    $step = $step_stmt->fetch(PDO::FETCH_ASSOC);
  1568→
  1569→    if ($submission && $step) {
  1570→        $subject = '[Renvoi] ' . ($submission['form_label'] ?? '') . ' — ' . ($step['label'] ?? '');
  1571→        $mail_sent = send_mail($old['email'], $subject, build_mail_html($submission, $step['label'], $new_token));
  1572→    }
  1573→
  1574→    app_log('token_regenerate', 'token:' . $old_token_id, 'Token régénéré pour ' . $old['email'] . ', nouveau token créé');
  1575→
  1576→    return [
  1577→        'success' => true,
  1578→        'message' => 'Nouveau lien de validation envoyé à ' . $old['email'],
  1579→    ];
  1580→}

  1761→/**
  1762→ * Gère l'upload d'un fichier pour une soumission
  1763→ *
  1764→ * @param array $file Le tableau $_FILES['field_name']
  1765→ * @param int $submission_id ID de la soumission
  1766→ * @param string $field_name Nom du champ
  1767→ * @return array ['success' => bool, 'message' => string, 'attachment_id' => int|null]
  1768→ */
  1769→function handle_file_upload(array $file, int $submission_id, string $field_name): array {
  1770→    // Vérifier les erreurs d'upload
  1771→    if ($file['error'] !== UPLOAD_ERR_OK) {
  1772→        $errors = [
  1773→            UPLOAD_ERR_INI_SIZE   => 'Le fichier dépasse la taille maximale autorisée par le serveur.',
  1774→            UPLOAD_ERR_FORM_SIZE  => 'Le fichier dépasse la taille maximale autorisée par le formulaire.',
  1775→            UPLOAD_ERR_PARTIAL    => 'Le fichier n\'a été que partiellement téléchargé.',
  1776→            UPLOAD_ERR_NO_FILE    => 'Aucun fichier n\'a été téléchargé.',
  1777→            UPLOAD_ERR_NO_TMP_DIR => 'Dossier temporaire manquant sur le serveur.',
  1778→            UPLOAD_ERR_CANT_WRITE => 'Erreur d\'écriture sur le serveur.',
  1779→        ];
  1780→        return ['success' => false, 'message' => $errors[$file['error']] ?? 'Erreur inconnue lors de l\'upload.', 'attachment_id' => null];
  1781→    }
  1782→
  1783→    // Vérifier la taille
  1784→    if ($file['size'] > get_max_file_size()) {
  1785→        return ['success' => false, 'message' => 'Le fichier dépasse la taille maximale autorisée (10 Mo).', 'attachment_id' => null];
  1786→    }
  1787→
  1788→    // Vérifier l'extension
  1789→    $ext = strtolower(pathinfo($file['name'], PATHINFO_EXTENSION));
  1790→    if (!in_array($ext, get_allowed_extensions())) {
  1791→        return ['success' => false, 'message' => 'Type de fichier non autorisé. Extensions acceptées : ' . implode(', ', get_allowed_extensions()) . '.', 'attachment_id' => null];
  1792→    }
  1793→
  1794→    // Vérifier le type MIME
  1795→    $finfo = finfo_open(FILEINFO_MIME_TYPE);
  1796→    $mime_type = finfo_file($finfo, $file['tmp_name']);
  1797→    finfo_close($finfo);
  1798→    if (!in_array($mime_type, get_allowed_mime_types())) {
  1799→        return ['success' => false, 'message' => 'Type MIME non autorisé : ' . $mime_type . '.', 'attachment_id' => null];
  1800→    }
  1801→
  1802→    // Lire le contenu du fichier pour stockage BLOB
  1803→    $file_content = file_get_contents($file['tmp_name']);
  1804→    if ($file_content === false) {
  1805→        return ['success' => false, 'message' => 'Erreur lors de la lecture du fichier.', 'attachment_id' => null];
  1806→    }
  1807→
  1808→    // Enregistrer dans la base de données avec le contenu BLOB
  1809→    $pdo = get_pdo();
  1810→    $pdo->prepare("INSERT INTO attachments (submission_id, field_name, original_name, stored_name, mime_type, file_size, file_data, uploaded_at) VALUES (?, ?, ?, '', ?, ?, ?, datetime('now'))")
  1811→        ->execute([$submission_id, $field_name, $file['name'], $mime_type, $file['size'], $file_content]);
  1812→
  1813→    $attachment_id = (int)$pdo->lastInsertId();
  1814→
  1815→    app_log('file_upload', 'submission:' . $submission_id, 'Fichier uploadé : ' . $file['name'] . ' (' . $mime_type . ', ' . $file['size'] . ' octets)');
  1816→
  1817→    return ['success' => true, 'message' => 'Fichier ' . $file['name'] . ' enregistré.', 'attachment_id' => $attachment_id];
  1818→}

  1876→/**
  1877→ * Délègue un token de validation à un autre validateur
  1878→ * L'ancien token est marqué comme traité (délégué) et un nouveau token est créé
  1879→ *
  1880→ * @param int $token_id ID du token à déléguer
  1881→ * @param string $to_email Email du délégataire
  1882→ * @param string $reason Motif de la délégation
  1883→ * @return array ['success' => bool, 'message' => string]
  1884→ */
  1885→function delegate_token(int $token_id, string $to_email, string $reason = ''): array {
  1886→    $pdo = get_pdo();
  1887→
  1888→    // Récupérer le token
  1889→    $stmt = $pdo->prepare("
  1890→        SELECT t.*, s.status as sub_status, s.data, f.label as form_label
  1891→        FROM tokens t
  1892→        JOIN submissions s ON s.id = t.submission_id
  1893→        JOIN forms f ON f.id = s.form_id
  1894→        WHERE t.id = ?
  1895→    ");
  1896→    $stmt->execute([$token_id]);
  1897→    $tok = $stmt->fetch(PDO::FETCH_ASSOC);
  1898→
  1899→    if (!$tok) {
  1900→        return ['success' => false, 'message' => 'Token introuvable.'];
  1901→    }
  1902→    if ($tok['done_at']) {
  1903→        return ['success' => false, 'message' => 'Ce token a déjà été traité.'];
  1904→    }
  1905→    if ($tok['sub_status'] !== 'en_cours') {
  1906→        return ['success' => false, 'message' => 'La soumission n\'est plus en cours.'];
  1907→    }
  1908→
  1909→    // Valider l'email du délégataire
  1910→    $to_email = strtolower(trim($to_email));
  1911→    if (!filter_var($to_email, FILTER_VALIDATE_EMAIL)) {
  1912→        return ['success' => false, 'message' => 'Adresse email invalide.'];
  1913→    }
  1914→    if ($to_email === $tok['email']) {
  1915→        return ['success' => false, 'message' => 'Vous ne pouvez pas déléguer à vous-même.'];
  1916→    }
  1917→
  1918→    // Vérifier qu'un token n'existe pas déjà pour cet email sur cette étape
  1919→    $dup_check = $pdo->prepare("SELECT 1 FROM tokens WHERE submission_id = ? AND step_id = ? AND email = ? AND done_at IS NULL");
  1920→    $dup_check->execute([$tok['submission_id'], $tok['step_id'], $to_email]);
  1921→    if ($dup_check->fetch()) {
  1922→        return ['success' => false, 'message' => 'Un token de validation est déjà actif pour ' . $to_email . ' sur cette étape.'];
  1923→    }
  1924→
  1925→    // Marquer l'ancien token comme traité (délégué)
  1926→    $pdo->prepare("UPDATE tokens SET done_at = datetime('now') WHERE id = ?")
  1927→        ->execute([$token_id]);
  1928→
  1929→    // Créer le nouveau token pour le délégataire
  1930→    $new_token = generate_token();
  1931→    $expire_days = (int)get_setting('token_expire_days', '30');
  1932→    $expires_at = date('Y-m-d H:i:s', strtotime("+{$expire_days} days"));
  1933→    $now = date('Y-m-d H:i:s');
  1934→
  1935→    $pdo->prepare("INSERT INTO tokens (submission_id, step_id, email, token, sent_at, expires_at) VALUES (?, ?, ?, ?, ?, ?)")
  1936→        ->execute([$tok['submission_id'], $tok['step_id'], $to_email, $new_token, $now, $expires_at]);
  1937→
  1938→    $new_token_id = (int)$pdo->lastInsertId();
  1939→
  1940→    // Enregistrer la délégation
  1941→    $pdo->prepare("INSERT INTO delegations (token_id, from_email, to_email, reason, delegated_at, new_token_id) VALUES (?, ?, ?, ?, datetime('now'), ?)")
  1942→        ->execute([$token_id, $tok['email'], $to_email, $reason, $new_token_id]);
  1943→
  1944→    // Envoyer l'email au délégataire
  1945→    $step_stmt = $pdo->prepare("SELECT label FROM steps WHERE id = ?");
  1946→    $step_stmt->execute([$tok['step_id']]);
  1947→    $step = $step_stmt->fetch(PDO::FETCH_ASSOC);
  1948→
  1949→    $submission = [
  1950→        'data' => $tok['data'],
  1951→        'form_label' => $tok['form_label'],
  1952→    ];
  1953→
  1954→    $subject = '[Délégation] ' . $tok['form_label'] . ' — ' . ($step['label'] ?? 'Validation requise');
  1955→    $mail_body = build_mail_html($submission, $step['label'] ?? 'Validation', $new_token);
  1956→
  1957→    // Ajouter un bloc de délégation en haut de l'email
  1958→    $delegation_notice = '<div style="background:#e8eaf6;border:1px solid #003189;border-radius:4px;padding:12px;margin-bottom:16px;">
  1959→        <strong>🔄 Délégation :</strong> Cette validation vous a été déléguée par <strong>' . h($tok['email']) . '</strong>.
  1960→        ' . (!empty($reason) ? '<br><em>Motif : ' . h($reason) . '</em>' : '') . '
  1961→    </div>';
  1962→    $mail_body = str_replace('<h2 style="color:#003189;">', $delegation_notice . '<h2 style="color:#003189;">', $mail_body);
  1963→
  1964→    send_mail($to_email, $subject, $mail_body);
  1965→
  1966→    // Notifier le délégateur que sa délégation a été prise en compte
  1967→    $confirm_subject = 'Délégation confirmée — ' . $tok['form_label'];
  1968→    $confirm_body = '<!DOCTYPE html><html lang="fr"><head><meta charset="UTF-8"></head>
  1969→<body style="font-family:Arial,sans-serif;max-width:600px;margin:auto;padding:20px;color:#222;">
  1970→  <h2 style="color:#003189;">🔄 Délégation confirmée</h2>
  1971→  <p>Votre validation pour <strong>' . h($tok['form_label']) . '</strong> (étape ' . h($step['label'] ?? '') . ') a été déléguée à <strong>' . h($to_email) . '</strong>.</p>
  1972→  <p>Vous n\'avez plus besoin d\'effectuer cette validation.</p>
  1973→  <p style="font-size:12px;color:#999;margin-top:24px;">Workflow DREETS — Ne pas répondre à cet email</p>
  1974→</body></html>';
  1975→    send_mail($tok['email'], $confirm_subject, $confirm_body);
  1976→
  1977→    app_log('token_delegate', 'token:' . $token_id, 'Token délégué de ' . $tok['email'] . ' à ' . $to_email . ($reason ? ' — Motif : ' . $reason : ''));
  1978→
  1979→    return ['success' => true, 'message' => 'Validation déléguée à ' . $to_email . '. Un email lui a été envoyé.'];
  1980→}
