   105→function generate_csrf_token(): string {
   106→    if (empty($_SESSION['csrf_token'])) {
   107→        $_SESSION['csrf_token'] = bin2hex(random_bytes(32));
   108→    }
   109→    return $_SESSION['csrf_token'];
   110→}
   111→
   112→function csrf_field(): string {
   113→    return '<input type="hidden" name="csrf_token" value="' . h(generate_csrf_token()) . '">';
   114→}

   116→function verify_csrf(): bool {
   117→    // Mode test : bypass CSRF
   118→    if (TEST_MODE) return true;
   119→    $token = $_POST['csrf_token'] ?? '';
   120→    return !empty($token) && hash_equals($_SESSION['csrf_token'] ?? '', $token);
   121→}

   599→/**
   600→ * Genere automatiquement un field_name a partir d'un libelle
   601→ * Ex: "Date de prise de poste" → "date_de_prise_de_poste"
   602→ * Ex: "Type d'arrivée" → "type_arrivee"
   603→ */
   604→function generate_field_name(string $label): string {
   605→    // Minuscules
   606→    $name = mb_strtolower($label, 'UTF-8');
   607→    // Supprimer les accents
   608→    $name = transliterator_transliterate('Any-Latin; Latin-ASCII; Lower()', $name);
   609→    if ($name === false) {
   610→        // Fallback manuel si intl pas dispo
   611→        $name = str_replace(
   612→            ['à','â','ä','é','è','ê','ë','ï','î','ô','ö','ù','û','ü','ç','œ','æ','ÿ'],
   613→            ['a','a','a','e','e','e','e','i','i','o','o','u','u','u','c','oe','ae','y'],
   614→            $name
   615→        );
   616→    }
   617→    // Remplacer tout ce qui n'est pas alphanumérique par un underscore
   618→    $name = preg_replace('/[^a-z0-9]+/', '_', $name);
   619→    // Nettoyer les underscores en double et en bordure
   620→    $name = trim($name, '_');
   621→    $name = preg_replace('/_+/', '_', $name);
   622→    return $name ?: 'champ';
   623→}

   625→/**
   626→ * Convertit une liste d'options (une par ligne) en JSON array
   627→ * Ex: "Option A\nOption B" → '["Option A","Option B"]'
   628→ * Si c'est déjà du JSON valide, le retourne tel quel
   629→ */
   630→function parse_options_input(string $input): ?string {
   631→    $input = trim($input);
   632→    if (empty($input)) return null;
   633→    
   634→    // Vérifier si c'est déjà du JSON valide
   635→    $decoded = json_decode($input, true);
   636→    if (json_last_error() === JSON_ERROR_NONE && is_array($decoded)) {
   637→        return $input;
   638→    }
   639→    
   640→    // Traiter comme une liste une option par ligne
   641→    $lines = array_filter(array_map('trim', explode("\n", $input)));
   642→    if (empty($lines)) return null;
   643→    
   644→    return json_encode($lines, JSON_UNESCAPED_UNICODE);
   645→}

   737→// ── MOTEUR WORKFLOW ───────────────────────────────────────────
   738→
   739→/**
   740→ * Déclenche la prochaine étape d'une soumission.
   741→ * Appelé à la création ET après chaque validation.
   742→ *
   743→ * Logique :
   744→ *  - Récupère toutes les étapes du formulaire triées par ordre
   745→ *  - Trouve le plus petit ordre sans tokens générés = prochaine étape
   746→ *  - Si ordre précédent non terminé (séquentiel) → on attend
   747→ *  - Génère les tokens pour tous les destinataires de l'étape courante
   748→ *  - Si plus aucune étape → soumission close
   749→ */
   750→function advance_workflow(int $submission_id): void {
   751→    $pdo = get_pdo();
   752→
   753→    $sub = $pdo->prepare("
   754→        SELECT s.*, f.label as form_label
   755→        FROM submissions s
   756→        JOIN forms f ON f.id = s.form_id
   757→        WHERE s.id = ?
   758→    ");
   759→    $sub->execute([$submission_id]);
   760→    $submission = $sub->fetch(PDO::FETCH_ASSOC);
   761→    if (!$submission || $submission['closed_at']) return;
   762→
   763→    // Toutes les étapes actives du formulaire
   764→    $steps = $pdo->prepare("
   765→        SELECT st.*, GROUP_CONCAT(sr.email, '|') as emails
   766→        FROM steps st
   767→        JOIN step_recipients sr ON sr.step_id = st.id
   768→        WHERE st.form_id = ? AND st.actif = 1
   769→        GROUP BY st.id
   770→        ORDER BY st.ordre ASC, st.id ASC
   771→    ");
   772→    $steps->execute([$submission['form_id']]);
   773→    $all_steps = $steps->fetchAll(PDO::FETCH_ASSOC);
   774→
   775→    if (empty($all_steps)) return;
   776→
   777→    // Tokens déjà générés pour cette soumission
   778→    $existing = $pdo->prepare("SELECT step_id, done_at FROM tokens WHERE submission_id = ?");
   779→    $existing->execute([$submission_id]);
   780→    $tokens_by_step = [];
   781→    foreach ($existing->fetchAll(PDO::FETCH_ASSOC) as $t) {
   782→        $tokens_by_step[$t['step_id']][] = $t['done_at'];
   783→    }
   784→
   785→    // Groupe les étapes par ordre
   786→    $by_ordre = [];
   787→    foreach ($all_steps as $step) {
   788→        $by_ordre[$step['ordre']][] = $step;
   789→    }
   790→    ksort($by_ordre);
   791→
   792→    foreach ($by_ordre as $ordre => $groupe) {
   793→        $step_ids    = array_column($groupe, 'id');
   794→        $all_started = count(array_intersect($step_ids, array_keys($tokens_by_step))) === count($step_ids);
   795→        $all_done    = $all_started && array_reduce($step_ids, function($carry, $sid) use ($tokens_by_step) {
   796→            if (!$carry) return false;
   797→            if (!isset($tokens_by_step[$sid])) return false;
   798→            foreach ($tokens_by_step[$sid] as $done) {
   799→                if (empty($done)) return false;
   800→            }
   801→            return true;
   802→        }, true);
   803→
   804→        if (!$all_started) {
   805→            // Cette étape n'a pas encore de tokens → on la démarre (parallèle dans le groupe)
   806→            $now = date('Y-m-d H:i:s');
   807→            $expire_days = (int)get_setting('token_expire_days', '30');
   808→            $expires_at = date('Y-m-d H:i:s', strtotime("+{$expire_days} days"));
   809→            foreach ($groupe as $step) {
   810→                $emails = explode('|', $step['emails']);
   811→                foreach ($emails as $email) {
   812→                    $token = generate_token();
   813→                    $pdo->prepare("INSERT INTO tokens (submission_id, step_id, email, token, sent_at, expires_at) VALUES (?,?,?,?,?,?)")
   814→                        ->execute([$submission_id, $step['id'], $email, $token, $now, $expires_at]);
   815→                    $subject = '[Action requise] ' . ($submission['form_label'] ?? '') . ' — ' . $step['label'];
   816→                    $mail_sent = send_mail($email, $subject, build_mail_html($submission, $step['label'], $token));
   817→                    if (!$mail_sent) {
   818→                        error_log("Workflow: mail failed for token $token to $email");
   819→                    }
   820→                }
   821→            }
   822→            return; // On attend que cette étape soit terminée avant de passer à la suivante
   823→        }
   824→
   825→        if (!$all_done) {
   826→            return; // Étape en cours, on attend
   827→        }
   828→
   829→        // Étape terminée → on continue la boucle vers l'ordre suivant
   830→    }
   831→
   832→    // Toutes les étapes sont terminées → on close et on notifie l'agent
   833→    $now = date('Y-m-d H:i:s');
   834→    $pdo->prepare("UPDATE submissions SET closed_at = ?, status = 'valide' WHERE id = ?")
   835→        ->execute([$now, $submission_id]);
   836→
   837→    // Notification de validation finale a l'agent
   838→    $agent_email = $submission['submitted_by'] ?? '';

   857→/**
   858→ * Appelé par validate.php quand un token est validé.
   859→ * Met à jour done_at puis avance le workflow.
   860→ */
   861→function validate_token(string $token, string $action = 'valider', string $comment = ''): array {
   862→    $pdo = get_pdo();
   863→
   864→    $row = $pdo->prepare("
   865→        SELECT t.*, st.label as step_label, s.form_id,
   866→               f.label as form_label, s.data, s.closed_at, s.status
   867→        FROM tokens t
   868→        JOIN steps st ON st.id = t.step_id
   869→        JOIN submissions s ON s.id = t.submission_id
   870→        JOIN forms f ON f.id = s.form_id
   871→        WHERE t.token = ?
   872→    ");
   873→    $row->execute([$token]);
   874→    $t = $row->fetch(PDO::FETCH_ASSOC);
   875→
   876→    if (!$t)             return ['status' => 'invalid'];
   877→    if ($t['done_at'])   return ['status' => 'already_done', 'data' => $t];
   878→    if ($t['closed_at']) return ['status' => 'closed',       'data' => $t];
   879→
   880→    // Vérifier si le token a expiré
   881→    if (!empty($t['expires_at']) && strtotime($t['expires_at']) < time()) {
   882→        return ['status' => 'expired', 'data' => $t];
   883→    }
   884→
   885→    // Récupérer les données actuelles
   886→    $data = json_decode($t['data'], true);
   887→    
   888→    // Ajouter la validation au tableau des validations
   889→    $validation = [
   890→        'step_label' => $t['step_label'],
   891→        'email' => $t['email'],
   892→        'action' => $action,
   893→        'commentaire' => $comment,
   894→        'date' => date('Y-m-d H:i:s')
   895→    ];
   896→    
   897→    // Initialiser le tableau des validations s'il n'existe pas
   898→    if (!isset($data['validations'])) {
   899→        $data['validations'] = [];
   900→    }
   901→    
   902→    // Ajouter la nouvelle validation
   903→    $data['validations'][] = $validation;
   904→    
   905→    // Mettre à jour les données avec les validations
   906→    $updated_data = json_encode($data);
   907→    
   908→    if ($action === 'refuser') {
   909→        // Pour le refus : mettre à jour done_at et fermer la soumission avec status refuse
   910→        $pdo->prepare("UPDATE tokens SET done_at = ? WHERE token = ?")
   911→            ->execute([date('Y-m-d H:i:s'), $token]);
   912→            
   913→        // Fermer la soumission avec le statut refuse
   914→        $pdo->prepare("UPDATE submissions SET closed_at = ?, status = 'refuse' WHERE id = ?")
   915→            ->execute([date('Y-m-d H:i:s'), $t['submission_id']]);
   916→
   917→        // Notifier l'agent du refus
   918→        $agent_email = $t['submitted_by'] ?? '';
   919→        if (!empty($agent_email) && filter_var($agent_email, FILTER_VALIDATE_EMAIL)) {
   920→            $refuse_subject = 'Demande refusée — ' . ($t['form_label'] ?? 'Workflow DREETS');
   921→            $refuse_body = '<!DOCTYPE html><html lang="fr"><head><meta charset="UTF-8"></head>
   922→<body style="font-family:Arial,sans-serif;max-width:600px;margin:auto;padding:20px;color:#222;">
   923→  <h2 style="color:#c0392b;">Demande refusée</h2>
   924→  <p>Votre demande <strong>' . h($t['form_label'] ?? '') . '</strong> a été refusée à l\'étape <strong>' . h($t['step_label']) . '</strong>.</p>
   925→  ' . (!empty($comment) ? '<p><strong>Motif :</strong> ' . h($comment) . '</p>' : '') . '
   926→  <p style="font-size:12px;color:#999;margin-top:24px;">Workflow DREETS — ' . h(get_setting('smtp_from', SMTP_FROM)) . '</p>
   927→</body></html>';
   928→            send_mail($agent_email, $refuse_subject, $refuse_body);
   929→        }
   930→    } else {
   931→        // Pour la validation : comportement normal
   932→        $pdo->prepare("UPDATE tokens SET done_at = ? WHERE token = ?")
   933→            ->execute([date('Y-m-d H:i:s'), $token]);
   934→
   935→        advance_workflow($t['submission_id']);
   936→    }
   937→    
   938→    // Mettre à jour les données de la soumission
   939→    $pdo->prepare("UPDATE submissions SET data = ? WHERE id = ?")
   940→        ->execute([$updated_data, $t['submission_id']]);
   941→
   942→    // Webhook notification
   943→    send_webhook('token_validated', ['submission_id' => $t['submission_id'], 'step_label' => $t['step_label'], 'email' => $t['email'], 'action' => $action]);
   944→
   945→    $t['done_at'] = date('Y-m-d H:i:s');
   946→    return ['status' => 'ok', 'data' => $t];
   947→}

  1274→// ── TOKEN REGENERATION ───────────────────────────────────────
  1275→
  1276→/**
  1277→ * Régénère un token expiré pour un validateur (admin uniquement)
  1278→ * Invalide l'ancien token et crée un nouveau avec une nouvelle date d'expiration
  1279→ *
  1280→ * @param int $old_token_id ID de l'ancien token
  1281→ * @return array ['success' => bool, 'message' => string]
  1282→ */
  1283→function regenerate_token(int $old_token_id): array {
  1284→    $pdo = get_pdo();
  1285→
  1286→    // Récupérer l'ancien token
  1287→    $stmt = $pdo->prepare("
  1288→        SELECT t.*, s.status as sub_status
  1289→        FROM tokens t
  1290→        JOIN submissions s ON s.id = t.submission_id
  1291→        WHERE t.id = ?
  1292→    ");
  1293→    $stmt->execute([$old_token_id]);
  1294→    $old = $stmt->fetch(PDO::FETCH_ASSOC);
  1295→
  1296→    if (!$old) {
  1297→        return ['success' => false, 'message' => 'Token introuvable.'];
  1298→    }
  1299→    if ($old['done_at']) {
  1300→        return ['success' => false, 'message' => 'Ce token a déjà été traité.'];
  1301→    }
  1302→    if ($old['sub_status'] !== 'en_cours') {
  1303→        return ['success' => false, 'message' => 'La soumission n\'est plus en cours.'];
  1304→    }
  1305→
  1306→    // Marquer l'ancien token comme traité (invalidé)
  1307→    $pdo->prepare("UPDATE tokens SET done_at = ? WHERE id = ?")
  1308→        ->execute([date('Y-m-d H:i:s'), $old_token_id]);
  1309→
  1310→    // Créer un nouveau token
  1311→    $new_token = generate_token();
  1312→    $expire_days = (int)get_setting('token_expire_days', '30');
  1313→    $expires_at = date('Y-m-d H:i:s', strtotime("+{$expire_days} days"));
  1314→    $now = date('Y-m-d H:i:s');

  1523→/**
  1524→ * Gère l'upload d'un fichier pour une soumission
  1525→ *
  1526→ * @param array $file Le tableau $_FILES['field_name']
  1527→ * @param int $submission_id ID de la soumission
  1528→ * @param string $field_name Nom du champ
  1529→ * @return array ['success' => bool, 'message' => string, 'attachment_id' => int|null]
  1530→ */
  1531→function handle_file_upload(array $file, int $submission_id, string $field_name): array {
  1532→    // Vérifier les erreurs d'upload
  1533→    if ($file['error'] !== UPLOAD_ERR_OK) {
  1534→        $errors = [
  1535→            UPLOAD_ERR_INI_SIZE   => 'Le fichier dépasse la taille maximale autorisée par le serveur.',
  1536→            UPLOAD_ERR_FORM_SIZE  => 'Le fichier dépasse la taille maximale autorisée par le formulaire.',
  1537→            UPLOAD_ERR_PARTIAL    => 'Le fichier n\'a été que partiellement téléchargé.',
  1538→            UPLOAD_ERR_NO_FILE    => 'Aucun fichier n\'a été téléchargé.',
  1539→            UPLOAD_ERR_NO_TMP_DIR => 'Dossier temporaire manquant sur le serveur.',
  1540→            UPLOAD_ERR_CANT_WRITE => 'Erreur d\'écriture sur le serveur.',
  1541→        ];
  1542→        return ['success' => false, 'message' => $errors[$file['error']] ?? 'Erreur inconnue lors de l\'upload.', 'attachment_id' => null];
  1543→    }
  1544→
  1545→    // Vérifier la taille
  1546→    if ($file['size'] > get_max_file_size()) {
  1547→        return ['success' => false, 'message' => 'Le fichier dépasse la taille maximale autorisée (10 Mo).', 'attachment_id' => null];
  1548→    }
  1549→
  1550→    // Vérifier l'extension
  1551→    $ext = strtolower(pathinfo($file['name'], PATHINFO_EXTENSION));
  1552→    if (!in_array($ext, get_allowed_extensions())) {
  1553→        return ['success' => false, 'message' => 'Type de fichier non autorisé. Extensions acceptées : ' . implode(', ', get_allowed_extensions()) . '.', 'attachment_id' => null];
  1554→    }
  1555→
  1556→    // Vérifier le type MIME
  1557→    $finfo = finfo_open(FILEINFO_MIME_TYPE);
  1558→    $mime_type = finfo_file($finfo, $file['tmp_name']);
  1559→    finfo_close($finfo);
  1560→    if (!in_array($mime_type, get_allowed_mime_types())) {
  1561→        return ['success' => false, 'message' => 'Type MIME non autorisé : ' . $mime_type . '.', 'attachment_id' => null];
  1562→    }
  1563→
  1564→    // Lire le contenu du fichier pour stockage BLOB
  1565→    $file_content = file_get_contents($file['tmp_name']);
  1566→    if ($file_content === false) {
  1567→        return ['success' => false, 'message' => 'Erreur lors de la lecture du fichier.', 'attachment_id' => null];
  1568→    }
  1569→
  1570→    // Enregistrer dans la base de données avec le contenu BLOB
  1571→    $pdo = get_pdo();
  1572→    $pdo->prepare("INSERT INTO attachments (submission_id, field_name, original_name, stored_name, mime_type, file_size, file_data, uploaded_at) VALUES (?, ?, ?, '', ?, ?, ?, datetime('now'))")
  1573→        ->execute([$submission_id, $field_name, $file['name'], $mime_type, $file['size'], $file_content]);
  1574→
  1575→    $attachment_id = (int)$pdo->lastInsertId();
  1576→
  1577→    app_log('file_upload', 'submission:' . $submission_id, 'Fichier uploadé : ' . $file['name'] . ' (' . $mime_type . ', ' . $file['size'] . ' octets)');
  1578→
  1579→    return ['success' => true, 'message' => 'Fichier ' . $file['name'] . ' enregistré.', 'attachment_id' => $attachment_id];
  1580→}

  1638→/**

  1641→ *
  1642→ * @param int $token_id ID du token à déléguer

  1647→function delegate_token(int $token_id, string $to_email, string $reason = ''): array {
  1648→    $pdo = get_pdo();
  1649→
  1650→    // Récupérer le token
  1651→    $stmt = $pdo->prepare("
  1652→        SELECT t.*, s.status as sub_status, s.data, f.label as form_label
  1653→        FROM tokens t
  1654→        JOIN submissions s ON s.id = t.submission_id
  1655→        JOIN forms f ON f.id = s.form_id
  1656→        WHERE t.id = ?
  1657→    ");

  1660→
  1661→    if (!$tok) {
  1662→        return ['success' => false, 'message' => 'Token introuvable.'];
  1663→    }
  1664→    if ($tok['done_at']) {
  1665→        return ['success' => false, 'message' => 'Ce token a déjà été traité.'];
  1666→    }
  1667→    if ($tok['sub_status'] !== 'en_cours') {
  1668→        return ['success' => false, 'message' => 'La soumission n\'est plus en cours.'];
  1669→    }
  1670→
  1671→    // Valider l'email du délégataire
  1672→    $to_email = strtolower(trim($to_email));
  1673→    if (!filter_var($to_email, FILTER_VALIDATE_EMAIL)) {
  1674→        return ['success' => false, 'message' => 'Adresse email invalide.'];
  1675→    }
  1676→    if ($to_email === $tok['email']) {
  1677→        return ['success' => false, 'message' => 'Vous ne pouvez pas déléguer à vous-même.'];
  1678→    }
  1679→
  1680→    // Vérifier qu'un token n'existe pas déjà pour cet email sur cette étape
  1681→    $dup_check = $pdo->prepare("SELECT 1 FROM tokens WHERE submission_id = ? AND step_id = ? AND email = ? AND done_at IS NULL");
  1682→    $dup_check->execute([$tok['submission_id'], $tok['step_id'], $to_email]);
  1683→    if ($dup_check->fetch()) {
  1684→        return ['success' => false, 'message' => 'Un token de validation est déjà actif pour ' . $to_email . ' sur cette étape.'];
  1685→    }
  1686→
  1687→    // Marquer l'ancien token comme traité (délégué)
  1688→    $pdo->prepare("UPDATE tokens SET done_at = datetime('now') WHERE id = ?")

  1695→    $now = date('Y-m-d H:i:s');

  1698→        ->execute([$tok['submission_id'], $tok['step_id'], $to_email, $new_token, $now, $expires_at]);
  1699→

  1701→
  1702→    // Enregistrer la délégation
  1703→    $pdo->prepare("INSERT INTO delegations (token_id, from_email, to_email, reason, delegated_at, new_token_id) VALUES (?, ?, ?, ?, datetime('now'), ?)")
  1704→        ->execute([$token_id, $tok['email'], $to_email, $reason, $new_token_id]);
  1705→
  1706→    // Envoyer l'email au délégataire
  1707→    $step_stmt = $pdo->prepare("SELECT label FROM steps WHERE id = ?");
  1708→    $step_stmt->execute([$tok['step_id']]);
  1709→    $step = $step_stmt->fetch(PDO::FETCH_ASSOC);
  1710→
  1711→    $submission = [
  1712→        'data' => $tok['data'],
  1713→        'form_label' => $tok['form_label'],
  1714→    ];
  1715→
  1716→    $subject = '[Délégation] ' . $tok['form_label'] . ' — ' . ($step['label'] ?? 'Validation requise');
  1717→    $mail_body = build_mail_html($submission, $step['label'] ?? 'Validation', $new_token);
  1718→
  1719→    // Ajouter un bloc de délégation en haut de l'email
  1720→    $delegation_notice = '<div style="background:#e8eaf6;border:1px solid #003189;border-radius:4px;padding:12px;margin-bottom:16px;">
  1721→        <strong>🔄 Délégation :</strong> Cette validation vous a été déléguée par <strong>' . h($tok['email']) . '</strong>.
  1722→        ' . (!empty($reason) ? '<br><em>Motif : ' . h($reason) . '</em>' : '') . '
  1723→    </div>';
  1724→    $mail_body = str_replace('<h2 style="color:#003189;">', $delegation_notice . '<h2 style="color:#003189;">', $mail_body);
