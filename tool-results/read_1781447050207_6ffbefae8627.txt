   134→// ── LAZY CRON ────────────────────────────────────────────────
   135→// Pas de cron sur le serveur. Le premier utilisateur qui se connecte
   136→// dans la journée/heure déclenche les tâches planifiées.
   137→
   138→/**
   139→ * Execute les tâches planifiées si nécessaire.
   140→ * Appelé automatiquement à chaque connexion (depuis get_pdo).
   141→ * 
   142→ * Fréquences :
   143→ * - daily  : alert_check (1x/jour)
   144→ * - hourly : remind (1x/heure)
   145→ */
   146→function run_lazy_cron(): void {
   147→    $pdo = get_pdo();
   148→    $now = time();
   149→    
   150→    $tasks = [
   151→        'remind'      => ['interval' => 3600,      'file' => __DIR__ . '/remind.php'],
   152→        'alert_check' => ['interval' => 86400,     'file' => __DIR__ . '/alert_check.php'],
   153→    ];
   154→    
   155→    foreach ($tasks as $key => $task) {
   156→        // Vérifier la dernière exécution
   157→        $stmt = $pdo->prepare("SELECT last_run FROM lazy_cron WHERE task_key = ?");
   158→        $stmt->execute([$key]);
   159→        $last_run = $stmt->fetchColumn();
   160→        
   161→        $should_run = false;
   162→        if ($last_run === false || $last_run === null) {
   163→            $should_run = true;
   164→        } else {
   165→            $last_ts = strtotime($last_run);
   166→            if (($now - $last_ts) >= $task['interval']) {
   167→                $should_run = true;
   168→            }
   169→        }
   170→        
   171→        if ($should_run) {
   172→            // Verrouillage : éviter les exécutions concurrentes
   173→            try {
   174→                $pdo->prepare("INSERT OR REPLACE INTO lazy_cron (task_key, last_run, run_count) VALUES (?, ?, COALESCE((SELECT run_count FROM lazy_cron WHERE task_key = ?), 0) + 1)")
   175→                    ->execute([$key, date('Y-m-d H:i:s', $now), $key]);
   176→            } catch (PDOException $e) {
   177→                continue; // Un autre processus est en cours
   178→            }
   179→            
   180→            // Exécuter la tâche dans un bloc try/catch pour ne pas casser la page
   181→            try {
   182→                // Les scripts remind.php et alert_check.php sont conçus pour être
   183→                // appelés en CLI. On les inclut directement avec output buffering.
   184→                ob_start();
   185→                require $task['file'];
   186→                ob_end_clean();
   187→            } catch (\Throwable $e) {
   188→                // Ne pas faire échouer la page utilisateur
   189→                error_log("Lazy cron error ({$key}): " . $e->getMessage());
   190→            }
   191→        }
   192→    }
   193→}

   243→/**
   244→ * Genere un UUID v4 (RFC 4122 compliant)
   245→ * Utilise random_bytes() pour la securite cryptographique
   246→ */
   247→function generate_uuid(): string {
   248→    $data = random_bytes(16);
   249→    $data[6] = chr(ord($data[6]) & 0x0f | 0x40); // version 4
   250→    $data[8] = chr(ord($data[8]) & 0x3f | 0x80); // variant RFC 4122
   251→    return vsprintf('%s%s-%s-%s-%s-%s%s%s', str_split(bin2hex($data), 4));
   252→}

  1048→/**
  1049→ * Genere automatiquement un field_name a partir d'un libelle
  1050→ * Ex: "Date de prise de poste" → "date_de_prise_de_poste"
  1051→ * Ex: "Type d'arrivée" → "type_arrivee"
  1052→ */
  1053→function generate_field_name(string $label): string {
  1054→    // Minuscules
  1055→    $name = mb_strtolower($label, 'UTF-8');
  1056→    // Supprimer les accents
  1057→    $name = transliterator_transliterate('Any-Latin; Latin-ASCII; Lower()', $name);
  1058→    if ($name === false) {
  1059→        // Fallback manuel si intl pas dispo
  1060→        $name = str_replace(
  1061→            ['à','â','ä','é','è','ê','ë','ï','î','ô','ö','ù','û','ü','ç','œ','æ','ÿ'],
  1062→            ['a','a','a','e','e','e','e','i','i','o','o','u','u','u','c','oe','ae','y'],
  1063→            $name
  1064→        );
  1065→    }
  1066→    // Remplacer tout ce qui n'est pas alphanumérique par un underscore
  1067→    $name = preg_replace('/[^a-z0-9]+/', '_', $name);
  1068→    // Nettoyer les underscores en double et en bordure
  1069→    $name = trim($name, '_');
  1070→    $name = preg_replace('/_+/', '_', $name);
  1071→    return $name ?: 'champ';
  1072→}

  1074→/**
  1075→ * Convertit une liste d'options (une par ligne) en JSON array
  1076→ * Ex: "Option A\nOption B" → '["Option A","Option B"]'
  1077→ * Si c'est déjà du JSON valide, le retourne tel quel
  1078→ */
  1079→function parse_options_input(string $input): ?string {
  1080→    $input = trim($input);
  1081→    if (empty($input)) return null;
  1082→    
  1083→    // Vérifier si c'est déjà du JSON valide
  1084→    $decoded = json_decode($input, true);
  1085→    if (json_last_error() === JSON_ERROR_NONE && is_array($decoded)) {
  1086→        return $input;
  1087→    }
  1088→    
  1089→    // Traiter comme une liste une option par ligne
  1090→    $lines = array_filter(array_map('trim', explode("\n", $input)));

  1188→/**
  1189→ * Déclenche la prochaine étape d'une soumission.
  1190→ * Appelé à la création ET après chaque validation.
  1191→ *
  1192→ * Logique :
  1193→ *  - Récupère toutes les étapes du formulaire triées par ordre
  1194→ *  - Trouve le plus petit ordre sans tokens générés = prochaine étape
  1195→ *  - Si ordre précédent non terminé (séquentiel) → on attend
  1196→ *  - Génère les tokens pour tous les destinataires de l'étape courante
  1197→ *  - Si plus aucune étape → soumission close
  1198→ */
  1199→function advance_workflow(int $submission_id): void {
  1200→    $pdo = get_pdo();
  1201→
  1202→    $sub = $pdo->prepare("
  1203→        SELECT s.*, f.label as form_label
  1204→        FROM submissions s
  1205→        JOIN forms f ON f.id = s.form_id
  1206→        WHERE s.id = ?
  1207→    ");
  1208→    $sub->execute([$submission_id]);
  1209→    $submission = $sub->fetch(PDO::FETCH_ASSOC);
  1210→    if (!$submission || $submission['closed_at']) return;
  1211→
  1212→    // Toutes les étapes actives du formulaire
  1213→    $steps = $pdo->prepare("
  1214→        SELECT st.*, GROUP_CONCAT(sr.email, '|') as emails
  1215→        FROM steps st
  1216→        JOIN step_recipients sr ON sr.step_id = st.id
  1217→        WHERE st.form_id = ? AND st.actif = 1
  1218→        GROUP BY st.id
  1219→        ORDER BY st.ordre ASC, st.id ASC
  1220→    ");
  1221→    $steps->execute([$submission['form_id']]);
  1222→    $all_steps = $steps->fetchAll(PDO::FETCH_ASSOC);
  1223→
  1224→    if (empty($all_steps)) return;
  1225→
  1226→    // Tokens déjà générés pour cette soumission
  1227→    $existing = $pdo->prepare("SELECT step_id, done_at FROM tokens WHERE submission_id = ?");
  1228→    $existing->execute([$submission_id]);
  1229→    $tokens_by_step = [];
  1230→    foreach ($existing->fetchAll(PDO::FETCH_ASSOC) as $t) {
  1231→        $tokens_by_step[$t['step_id']][] = $t['done_at'];
  1232→    }
  1233→
  1234→    // Groupe les étapes par ordre
  1235→    $by_ordre = [];
  1236→    foreach ($all_steps as $step) {
  1237→        $by_ordre[$step['ordre']][] = $step;
  1238→    }
  1239→    ksort($by_ordre);
  1240→
  1241→    foreach ($by_ordre as $ordre => $groupe) {
  1242→        $step_ids    = array_column($groupe, 'id');
  1243→        $all_started = count(array_intersect($step_ids, array_keys($tokens_by_step))) === count($step_ids);
  1244→        $all_done    = $all_started && array_reduce($step_ids, function($carry, $sid) use ($tokens_by_step) {
  1245→            if (!$carry) return false;
  1246→            if (!isset($tokens_by_step[$sid])) return false;
  1247→            foreach ($tokens_by_step[$sid] as $done) {
  1248→                if (empty($done)) return false;
  1249→            }
  1250→            return true;
  1251→        }, true);
  1252→
  1253→        if (!$all_started) {
  1254→            // Cette étape n'a pas encore de tokens → on la démarre (parallèle dans le groupe)
  1255→            $now = date('Y-m-d H:i:s');
  1256→            $expire_days = (int)get_setting('token_expire_days', '30');
  1257→            $expires_at = date('Y-m-d H:i:s', strtotime("+{$expire_days} days"));
  1258→            foreach ($groupe as $step) {
  1259→                $emails = explode('|', $step['emails']);
  1260→                foreach ($emails as $email) {
  1261→                    $token = generate_token();
  1262→                    $pdo->prepare("INSERT INTO tokens (submission_id, step_id, email, token, sent_at, expires_at) VALUES (?,?,?,?,?,?)")
  1263→                        ->execute([$submission_id, $step['id'], $email, $token, $now, $expires_at]);
  1264→                    $subject = '[Action requise] ' . ($submission['form_label'] ?? '') . ' — ' . $step['label'];
  1265→                    $mail_sent = send_mail($email, $subject, build_mail_html($submission, $step['label'], $token));
  1266→                    if (!$mail_sent) {
  1267→                        error_log("Workflow: mail failed for token $token to $email");
  1268→                    }
  1269→                }
  1270→            }
  1271→            return; // On attend que cette étape soit terminée avant de passer à la suivante
  1272→        }
  1273→
  1274→        if (!$all_done) {
  1275→            return; // Étape en cours, on attend
  1276→        }
  1277→
  1278→        // Étape terminée → on continue la boucle vers l'ordre suivant
  1279→    }
  1280→
  1281→    // Toutes les étapes sont terminées → on close et on notifie l'agent
  1282→    $now = date('Y-m-d H:i:s');
  1283→    $pdo->prepare("UPDATE submissions SET closed_at = ?, status = 'valide' WHERE id = ?")
  1284→        ->execute([$now, $submission_id]);
  1285→
  1286→    // Notification de validation finale a l'agent
  1287→    $agent_email = $submission['submitted_by'] ?? '';
  1288→    if (!empty($agent_email) && filter_var($agent_email, FILTER_VALIDATE_EMAIL)) {
  1289→        $subject = 'Demande validée — ' . ($submission['form_label'] ?? 'Workflow DREETS');
  1290→        $body = '<!DOCTYPE html><html lang="fr"><head><meta charset="UTF-8"></head>
  1291→<body style="font-family:Arial,sans-serif;max-width:600px;margin:auto;padding:20px;color:#222;">
  1292→  <h2 style="color:#1a6b3c;">✓ Demande validée</h2>
  1293→  <p>Votre demande <strong>' . h($submission['form_label'] ?? '') . '</strong> a été <strong>validée</strong> par l\'ensemble des validateurs.</p>
  1294→  <p>Le processus de workflow est désormais terminé.</p>
  1295→  <p style="font-size:12px;color:#999;margin-top:24px;">Workflow DREETS — ' . h(get_setting('smtp_from', SMTP_FROM)) . '</p>
  1296→</body></html>';
  1297→        send_mail($agent_email, $subject, $body);
  1298→    }
  1299→
  1300→    app_log('workflow_complete', 'submission:' . $submission_id, 'Formulaire ' . ($submission['form_label'] ?? '') . ' validé', $agent_email);
  1301→
  1302→    // Webhook notification
  1303→    send_webhook('workflow_complete', ['submission_id' => $submission_id, 'form_label' => $submission['form_label'] ?? '', 'submitted_by' => $submission['submitted_by'] ?? '']);
  1304→}

  1306→/**
  1307→ * Appelé par validate.php quand un token est validé.
  1308→ * Met à jour done_at puis avance le workflow.
  1309→ */
  1310→function validate_token(string $token, string $action = 'valider', string $comment = ''): array {
  1311→    $pdo = get_pdo();
  1312→
  1313→    $row = $pdo->prepare("
  1314→        SELECT t.*, st.label as step_label, s.form_id,
  1315→               f.label as form_label, s.data, s.closed_at, s.status
  1316→        FROM tokens t
  1317→        JOIN steps st ON st.id = t.step_id
  1318→        JOIN submissions s ON s.id = t.submission_id
  1319→        JOIN forms f ON f.id = s.form_id
  1320→        WHERE t.token = ?
  1321→    ");
  1322→    $row->execute([$token]);
  1323→    $t = $row->fetch(PDO::FETCH_ASSOC);
  1324→
  1325→    if (!$t)             return ['status' => 'invalid'];
  1326→    if ($t['done_at'])   return ['status' => 'already_done', 'data' => $t];
  1327→    if ($t['closed_at']) return ['status' => 'closed',       'data' => $t];
  1328→
  1329→    // Vérifier si le token a expiré
  1330→    if (!empty($t['expires_at']) && strtotime($t['expires_at']) < time()) {
  1331→        return ['status' => 'expired', 'data' => $t];
  1332→    }
  1333→
  1334→    // Récupérer les données actuelles
  1335→    $data = json_decode($t['data'], true);
  1336→    
  1337→    // Ajouter la validation au tableau des validations
  1338→    $validation = [
  1339→        'step_label' => $t['step_label'],
  1340→        'email' => $t['email'],
  1341→        'action' => $action,
  1342→        'commentaire' => $comment,
  1343→        'date' => date('Y-m-d H:i:s')
  1344→    ];
  1345→    
  1346→    // Initialiser le tableau des validations s'il n'existe pas
  1347→    if (!isset($data['validations'])) {
  1348→        $data['validations'] = [];
  1349→    }
  1350→    
  1351→    // Ajouter la nouvelle validation
  1352→    $data['validations'][] = $validation;
  1353→    
  1354→    // Mettre à jour les données avec les validations
  1355→    $updated_data = json_encode($data);
  1356→    
  1357→    if ($action === 'refuser') {
  1358→        // Pour le refus : mettre à jour done_at et fermer la soumission avec status refuse
  1359→        $pdo->prepare("UPDATE tokens SET done_at = ? WHERE token = ?")
  1360→            ->execute([date('Y-m-d H:i:s'), $token]);
  1361→            
  1362→        // Fermer la soumission avec le statut refuse
  1363→        $pdo->prepare("UPDATE submissions SET closed_at = ?, status = 'refuse' WHERE id = ?")
  1364→            ->execute([date('Y-m-d H:i:s'), $t['submission_id']]);
  1365→
  1366→        // Notifier l'agent du refus
  1367→        $agent_email = $t['submitted_by'] ?? '';
  1368→        if (!empty($agent_email) && filter_var($agent_email, FILTER_VALIDATE_EMAIL)) {
  1369→            $refuse_subject = 'Demande refusée — ' . ($t['form_label'] ?? 'Workflow DREETS');
  1370→            $refuse_body = '<!DOCTYPE html><html lang="fr"><head><meta charset="UTF-8"></head>
  1371→<body style="font-family:Arial,sans-serif;max-width:600px;margin:auto;padding:20px;color:#222;">
  1372→  <h2 style="color:#c0392b;">Demande refusée</h2>
  1373→  <p>Votre demande <strong>' . h($t['form_label'] ?? '') . '</strong> a été refusée à l\'étape <strong>' . h($t['step_label']) . '</strong>.</p>
  1374→  ' . (!empty($comment) ? '<p><strong>Motif :</strong> ' . h($comment) . '</p>' : '') . '
  1375→  <p style="font-size:12px;color:#999;margin-top:24px;">Workflow DREETS — ' . h(get_setting('smtp_from', SMTP_FROM)) . '</p>
  1376→</body></html>';
  1377→            send_mail($agent_email, $refuse_subject, $refuse_body);
  1378→        }
  1379→    } else {
  1380→        // Pour la validation : comportement normal
  1381→        $pdo->prepare("UPDATE tokens SET done_at = ? WHERE token = ?")
  1382→            ->execute([date('Y-m-d H:i:s'), $token]);
  1383→
  1384→        advance_workflow($t['submission_id']);
  1385→    }
  1386→    
  1387→    // Mettre à jour les données de la soumission
  1388→    $pdo->prepare("UPDATE submissions SET data = ? WHERE id = ?")
  1389→        ->execute([$updated_data, $t['submission_id']]);
  1390→
  1391→    // Webhook notification
  1392→    send_webhook('token_validated', ['submission_id' => $t['submission_id'], 'step_label' => $t['step_label'], 'email' => $t['email'], 'action' => $action]);
  1393→
  1394→    $t['done_at'] = date('Y-m-d H:i:s');
  1395→    return ['status' => 'ok', 'data' => $t];
  1396→}

  1725→/**
  1726→ * Régénère un token expiré pour un validateur (admin uniquement)
  1727→ * Invalide l'ancien token et crée un nouveau avec une nouvelle date d'expiration
  1728→ *
  1729→ * @param int $old_token_id ID de l'ancien token
  1730→ * @return array ['success' => bool, 'message' => string]
  1731→ */
  1732→function regenerate_token(int $old_token_id): array {
  1733→    $pdo = get_pdo();
  1734→
  1735→    // Récupérer l'ancien token
  1736→    $stmt = $pdo->prepare("
  1737→        SELECT t.*, s.status as sub_status
  1738→        FROM tokens t
  1739→        JOIN submissions s ON s.id = t.submission_id
  1740→        WHERE t.id = ?
  1741→    ");
  1742→    $stmt->execute([$old_token_id]);
  1743→    $old = $stmt->fetch(PDO::FETCH_ASSOC);
  1744→
  1745→    if (!$old) {
  1746→        return ['success' => false, 'message' => 'Token introuvable.'];
  1747→    }
  1748→    if ($old['done_at']) {
  1749→        return ['success' => false, 'message' => 'Ce token a déjà été traité.'];
  1750→    }
  1751→    if ($old['sub_status'] !== 'en_cours') {
  1752→        return ['success' => false, 'message' => 'La soumission n\'est plus en cours.'];
  1753→    }
  1754→
  1755→    // Marquer l'ancien token comme traité (invalidé)
  1756→    $pdo->prepare("UPDATE tokens SET done_at = ? WHERE id = ?")
  1757→        ->execute([date('Y-m-d H:i:s'), $old_token_id]);
  1758→
  1759→    // Créer un nouveau token
  1760→    $new_token = generate_token();
  1761→    $expire_days = (int)get_setting('token_expire_days', '30');
  1762→    $expires_at = date('Y-m-d H:i:s', strtotime("+{$expire_days} days"));
  1763→    $now = date('Y-m-d H:i:s');
  1764→
  1765→    $pdo->prepare("INSERT INTO tokens (submission_id, step_id, email, token, sent_at, expires_at) VALUES (?, ?, ?, ?, ?, ?)")
  1766→        ->execute([$old['submission_id'], $old['step_id'], $old['email'], $new_token, $now, $expires_at]);
  1767→
  1768→    // Envoyer le nouveau lien par email
  1769→    $sub_stmt = $pdo->prepare("
  1770→        SELECT s.*, f.label as form_label FROM submissions s
  1771→        JOIN forms f ON f.id = s.form_id WHERE s.id = ?
  1772→    ");
  1773→    $sub_stmt->execute([$old['submission_id']]);
  1774→    $submission = $sub_stmt->fetch(PDO::FETCH_ASSOC);
  1775→
  1776→    $step_stmt = $pdo->prepare("SELECT label FROM steps WHERE id = ?");
  1777→    $step_stmt->execute([$old['step_id']]);
  1778→    $step = $step_stmt->fetch(PDO::FETCH_ASSOC);
  1779→
  1780→    if ($submission && $step) {
  1781→        $subject = '[Renvoi] ' . ($submission['form_label'] ?? '') . ' — ' . ($step['label'] ?? '');
  1782→        $mail_sent = send_mail($old['email'], $subject, build_mail_html($submission, $step['label'], $new_token));
  1783→    }
  1784→
  1785→    app_log('token_regenerate', 'token:' . $old_token_id, 'Token régénéré pour ' . $old['email'] . ', nouveau token créé');
  1786→
  1787→    return [
  1788→        'success' => true,
  1789→        'message' => 'Nouveau lien de validation envoyé à ' . $old['email'],
  1790→    ];
  1791→}
